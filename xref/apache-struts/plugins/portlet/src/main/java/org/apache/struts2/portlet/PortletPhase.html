1   package org.apache.struts2.portlet;
2   
3   /**
4    * Defines phases of portlet processing per the portlet specification.
5    */
6   public enum PortletPhase {
7   
8       /**
9        * Constant used for the render phase (
10       * {@link javax.portlet.Portlet#render(javax.portlet.RenderRequest, javax.portlet.RenderResponse)})
11       */
12      RENDER_PHASE,
13  
14      /**
15       * Constant used for the action phase (
16       * {@link javax.portlet.Portlet#processAction(javax.portlet.ActionRequest, javax.portlet.ActionResponse)})
17       */
18      ACTION_PHASE,
19  
20      /**
21       * Constant used for the event phase
22       */
23      EVENT_PHASE,
24  
25      /**
26       * Constant used for the serve resource phase that was added with the 2.0 portlet specification.
27       */
28      SERVE_RESOURCE_PHASE;
29  
30      public boolean isRender() {
31          return this.equals(RENDER_PHASE);
32      }
33  
34      public boolean isAction() {
35          return this.equals(ACTION_PHASE);
36      }
37  
38      public boolean isEvent() {
39          return this.equals(EVENT_PHASE);
40      }
41  
42      public boolean isResource() {
43          return this.equals(SERVE_RESOURCE_PHASE);
44      }
45  
46  }
