4   package com.opensymphony.xwork2.spring;
5   
6   import com.opensymphony.xwork2.*;
7   import com.opensymphony.xwork2.config.providers.XmlConfigurationProvider;
8   import org.springframework.context.ApplicationContext;
9   
10  /**
11   * Test loading actions from the Spring Application Context.
12   *
13   * @author Simon Stewart
14   */
15  public class ActionsFromSpringTest extends XWorkTestCase {
16      private ApplicationContext appContext;
17  
18      @Override public void setUp() throws Exception {
19          super.setUp();
20  
21          // Set up XWork
22          XmlConfigurationProvider provider = new XmlConfigurationProvider("com/opensymphony/xwork2/spring/actionContext-xwork.xml");
23          container.inject(provider);
24          loadConfigurationProviders(provider);
25          appContext = ((SpringObjectFactory)container.getInstance(ObjectFactory.class)).appContext;
26      }
27  
28      public void testLoadSimpleAction() throws Exception {
29          ActionProxy proxy = actionProxyFactory.createActionProxy(null, "simpleAction", null, null);
30          Object action = proxy.getAction();
31  
32          Action expected = (Action) appContext.getBean("simple-action");
33  
34          assertEquals(expected.getClass(), action.getClass());
35      }
36  
37      public void testLoadActionWithDependencies() throws Exception {
38          ActionProxy proxy = actionProxyFactory.createActionProxy(null, "dependencyAction", null, null);
39          SimpleAction action = (SimpleAction) proxy.getAction();
40  
41          assertEquals("injected", action.getBlah());
42      }
43  
44      public void testProxiedActionIsNotStateful() throws Exception {
45          ActionProxy proxy = actionProxyFactory.createActionProxy(null, "proxiedAction", null, null);
46          SimpleAction action = (SimpleAction) proxy.getAction();
47  
48          action.setBlah("Hello World");
49  
50          proxy = actionProxyFactory.createActionProxy(null, "proxiedAction", null, null);
51          action = (SimpleAction) proxy.getAction();
52  
53          // If the action is a singleton, this test will fail
54          SimpleAction sa = new SimpleAction();
55          assertEquals(sa.getBlah(), action.getBlah());
56  
57          // And if the advice is not being applied, this will be SUCCESS.
58          String result = action.execute();
59          assertEquals(Action.INPUT, result);
60      }
61  
62      public void testAutoProxiedAction() throws Exception {
63          ActionProxy proxy = actionProxyFactory.createActionProxy(null, "autoProxiedAction", null, null);
64  
65          SimpleAction action = (SimpleAction) proxy.getAction();
66  
67          String result = action.execute();
68          assertEquals(Action.INPUT, result);
69      }
70      
71      public void testActionWithSpringResult() throws Exception {
72      	        ActionProxy proxy = actionProxyFactory.createActionProxy(null, "simpleActionSpringResult", null, null);
73      	                
74      	        proxy.execute();
75      	        
76      	        SpringResult springResult = (SpringResult) proxy.getInvocation().getResult();
77      	        assertTrue(springResult.isInitialize());
78      	        assertNotNull(springResult.getStringParameter());
79      }
80  }
