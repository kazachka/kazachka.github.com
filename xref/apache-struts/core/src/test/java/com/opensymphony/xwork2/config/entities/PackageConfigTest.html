16  package com.opensymphony.xwork2.config.entities;
17  
18  import com.opensymphony.xwork2.XWorkTestCase;
19  
20  public class PackageConfigTest extends XWorkTestCase {
21  
22      public void testFullDefaultInterceptorRef() {
23          PackageConfig cfg1 = new PackageConfig.Builder("pkg1")
24                  .defaultInterceptorRef("ref1").build();
25          PackageConfig cfg2 = new PackageConfig.Builder("pkg2").defaultInterceptorRef("ref2").build();
26          PackageConfig cfg = new PackageConfig.Builder("pkg")
27                  .addParent(cfg1)
28                  .addParent(cfg2)
29                  .build();
30          
31          assertEquals("ref2", cfg.getFullDefaultInterceptorRef());
32      }
33  
34      public void testStrictDMIInheritance() {
35          // given
36          PackageConfig parent = new PackageConfig.Builder("parent").build();
37  
38          // when
39          PackageConfig child = new PackageConfig.Builder("child")
40                  .addParent(parent)
41                  .build();
42  
43          // then
44          assertTrue(child.isStrictMethodInvocation());
45      }
46  
47      public void testStrictDMIInheritanceDisabledInParentPackage() {
48          // given
49          PackageConfig parent = new PackageConfig.Builder("parent")
50                  .strictMethodInvocation(false)
51                  .build();
52  
53          // when
54          PackageConfig child = new PackageConfig.Builder("child")
55                  .addParent(parent)
56                  .build();
57  
58          // then
59          assertTrue(child.isStrictMethodInvocation());
60      }
61  
62      public void testStrictDMIInheritanceDisabledInBothPackage() {
63          // given
64          PackageConfig parent = new PackageConfig.Builder("parent")
65                  .strictMethodInvocation(false)
66                  .build();
67  
68          // when
69          PackageConfig child = new PackageConfig.Builder("child")
70                  .addParent(parent)
71                  .strictMethodInvocation(false)
72                  .build();
73  
74          // then
75          assertFalse(child.isStrictMethodInvocation());
76      }
77  
78      public void testStrictDMIInheritanceDisabledInChildPackage() {
