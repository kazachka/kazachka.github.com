3   import java.lang.reflect.InvocationHandler;
4   import java.lang.reflect.Method;
5   import java.lang.reflect.Proxy;
6   import java.util.Map;
7   
8   /**
9    * ObjectFactory that returns a FooProxy in the buildBean if the clazz is FooAction 
10   */
11  public class ProxyObjectFactory extends ObjectFactory {
12  
13      /**
14       * It returns an instance of the bean except if the class is FooAction. 
15       * In this case, it returns a FooProxy of it.
16       */
17      @Override
18      public Object buildBean(Class clazz, Map<String, Object> extraContext)
19          throws Exception {
20          Object bean = super.buildBean(clazz, extraContext);
21          if(clazz.equals(ProxyInvocationAction.class)) {
22              return Proxy.newProxyInstance(bean.getClass()
23                  .getClassLoader(), bean.getClass().getInterfaces(),
24                  new ProxyInvocationProxy(bean));
25  
26          }
27          return bean;
28      }
29      
30      /**
31       * Simple proxy that just invokes the method on the target on the invoke method
32       */
33      public class ProxyInvocationProxy implements InvocationHandler {
34  
35          private Object target;
36  
37          public ProxyInvocationProxy(Object target) {
38              this.target = target;
39          }
40  
41          public Object invoke(Object proxy, Method m, Object[] args)
42              throws Throwable {
43              return m.invoke(target, args);
44          }
45      }
46  }
