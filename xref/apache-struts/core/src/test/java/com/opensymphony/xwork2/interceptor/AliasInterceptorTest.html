28  /**
29   * AliasInterceptorTest
30   *
31   * <p>
32   * Test of aliasInterceptor specifically depends on actionTest test defined in /test/xwork.xml
33   * stack.getContext().putAll(params);
34   * </p>
35   * e.g.
36   * <pre>
37   * <action name="aliasTest" class="com.opensymphony.xwork2.SimpleAction">
38   *    <param name="aliases">#{ "aliasSource" : "aliasDest", "bar":"baz" }</param>
39   *    <interceptor-ref name="defaultStack"/>
40   *    <interceptor-ref name="alias"/>
41   * </action>
42   * </pre>
43   * @author Matthew Payne
44   */
45  public class AliasInterceptorTest extends XWorkTestCase {
46  
47      public void testUsingDefaultInterceptorThatAliasPropertiesAreCopied() throws Exception {
48          Map<String, Object> params = new HashMap<>();
49          params.put("aliasSource", "source here");
50  
51          XmlConfigurationProvider provider = new XmlConfigurationProvider("xwork-sample.xml");
52          container.inject(provider);
53          loadConfigurationProviders(provider);
54          ActionProxy proxy = actionProxyFactory.createActionProxy("", "aliasTest", null, params);
55          SimpleAction actionOne = (SimpleAction) proxy.getAction();
56          actionOne.setAliasSource("name to be copied");
57          actionOne.setFoo(17);
58          actionOne.setBar(23);
59          proxy.execute();
60          assertEquals(actionOne.getAliasSource(), actionOne.getAliasDest());
61      }

63      public void testInvalidAliasExpression() throws Exception {
64          Action action = new SimpleFooAction();
65          MockActionInvocation mai = new MockActionInvocation();
66  
67          MockActionProxy map = new MockActionProxy();
68  
69          ActionConfig cfg = new ActionConfig.Builder("", "", "")
70                  .addParam("aliases", "invalid alias expression")
71                  .build();
72          map.setConfig(cfg);
73  
74          mai.setProxy(map);
75          mai.setAction(action);
76          mai.setInvocationContext(ActionContext.getContext());
77  
78          AliasInterceptor ai = new AliasInterceptor();
79          ai.init();
80  
81          ai.intercept(mai);
82  
83          ai.destroy();
84      }
