6   import com.opensymphony.xwork2.XWorkTestCase;
7   import com.opensymphony.xwork2.validator.DelegatingValidatorContext;
8   import com.opensymphony.xwork2.validator.ValidatorContext;
9   import com.opensymphony.xwork2.validator.VisitorValidatorTestAction;
10  import com.opensymphony.xwork2.validator.validators.VisitorFieldValidator.AppendingValidatorContext;
11  
12  public class AppendingValidatorContextTest extends XWorkTestCase {
13  
14      private static final String FIRST_NAME = "first";
15      private static final String SECOND_NAME = "second";
16      private static final String FIELD_NAME = "fieldName";
17      private static final String FULL_FIELD_NAME = FIRST_NAME + "." + SECOND_NAME + "." + FIELD_NAME;
18  
19      private VisitorValidatorTestAction action;
20      private VisitorFieldValidator.AppendingValidatorContext validatorContext;
21  
22      @Override
23      protected void setUp() throws Exception {
24          super.setUp();
25  
26          action = new VisitorValidatorTestAction();
27          ValidatorContext vc1 = new DelegatingValidatorContext(action);
28  
29          VisitorFieldValidator.AppendingValidatorContext vc2 = new AppendingValidatorContext(
30                  vc1, "value", FIRST_NAME, "");
31          validatorContext = new AppendingValidatorContext(vc2, "value", SECOND_NAME, "");
32      }
33  
34      public void testGetFullFieldName() throws Exception {
35          String fullFieldName = validatorContext.getFullFieldName(FIELD_NAME);
36          assertEquals(FULL_FIELD_NAME, fullFieldName);
37      }
38  
39      public void testAddFieldError() throws Exception {
40          validatorContext.addFieldError(FIELD_NAME, "fieldError");
41          assertTrue(action.hasFieldErrors());
42  
43          Map<String, List<String>> fieldErrors = action.getFieldErrors();
44          assertEquals(1, fieldErrors.size());
45          assertTrue(fieldErrors.containsKey(FULL_FIELD_NAME));
46      }
47  }
