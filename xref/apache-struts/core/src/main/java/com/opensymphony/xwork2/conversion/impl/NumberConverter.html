12  public class NumberConverter extends DefaultTypeConverter {

14      public Object convertValue(Map<String, Object> context, Object target, Member member, String propertyName, Object value, Class toType) {
15          if (value instanceof String) {
16              if (toType == BigDecimal.class) {
17                  return new BigDecimal((String) value);
18              } else if (toType == BigInteger.class) {
19                  return new BigInteger((String) value);
20              } else if (toType.isPrimitive()) {
21                  Object convertedValue = super.convertValue(context, value, toType);
22                  String stringValue = (String) value;
23                  if (!isInRange((Number) convertedValue, stringValue, toType))
24                      throw new XWorkException("Overflow or underflow casting: \"" + stringValue + "\" into class " + convertedValue.getClass().getName());
25  
26                  return convertedValue;
27              } else {
28                  String stringValue = (String) value;
29                  if (!toType.isPrimitive() && (stringValue == null || stringValue.length() == 0)) {
30                      return null;
31                  }
32                  NumberFormat numFormat = NumberFormat.getInstance(getLocale(context));
33                  ParsePosition parsePos = new ParsePosition(0);
34                  if (isIntegerType(toType)) {
35                      numFormat.setParseIntegerOnly(true);
36                  }
37                  numFormat.setGroupingUsed(true);
38                  Number number = numFormat.parse(stringValue, parsePos);
39  
40                  if (parsePos.getIndex() != stringValue.length()) {
41                      throw new XWorkException("Unparseable number: \"" + stringValue + "\" at position "
42                              + parsePos.getIndex());
43                  } else {
44                      if (!isInRange(number, stringValue, toType))
45                          throw new XWorkException("Overflow or underflow casting: \"" + stringValue + "\" into class " + number.getClass().getName());
46  
47                      value = super.convertValue(context, number, toType);
48                  }
49              }
50          } else if (value instanceof Object[]) {
51              Object[] objArray = (Object[]) value;
52  
53              if (objArray.length == 1) {
54                  return convertValue(context, null, null, null, objArray[0], toType);
55              }
56          }
57  
58          // pass it through DefaultTypeConverter
59          return super.convertValue(context, value, toType);
60      }

62      protected boolean isInRange(Number value, String stringValue, Class toType) {
63          Number bigValue = null;
64          Number lowerBound = null;
65          Number upperBound = null;
66  
67          try {
68              if (double.class == toType || Double.class == toType) {
69                  bigValue = new BigDecimal(stringValue);
70                  // Double.MIN_VALUE is the smallest positive non-zero number
71                  lowerBound = BigDecimal.valueOf(Double.MAX_VALUE).negate();
72                  upperBound = BigDecimal.valueOf(Double.MAX_VALUE);
73              } else if (float.class == toType || Float.class == toType) {
74                  bigValue = new BigDecimal(stringValue);
75                  // Float.MIN_VALUE is the smallest positive non-zero number
76                  lowerBound = BigDecimal.valueOf(Float.MAX_VALUE).negate();
77                  upperBound = BigDecimal.valueOf(Float.MAX_VALUE);
78              } else if (byte.class == toType || Byte.class == toType) {
79                  bigValue = new BigInteger(stringValue);
80                  lowerBound = BigInteger.valueOf(Byte.MIN_VALUE);
81                  upperBound = BigInteger.valueOf(Byte.MAX_VALUE);
82              } else if (char.class == toType || Character.class == toType) {
83                  bigValue = new BigInteger(stringValue);
84                  lowerBound = BigInteger.valueOf(Character.MIN_VALUE);
85                  upperBound = BigInteger.valueOf(Character.MAX_VALUE);
86              } else if (short.class == toType || Short.class == toType) {
87                  bigValue = new BigInteger(stringValue);
88                  lowerBound = BigInteger.valueOf(Short.MIN_VALUE);
89                  upperBound = BigInteger.valueOf(Short.MAX_VALUE);
90              } else if (int.class == toType || Integer.class == toType) {
91                  bigValue = new BigInteger(stringValue);
92                  lowerBound = BigInteger.valueOf(Integer.MIN_VALUE);
93                  upperBound = BigInteger.valueOf(Integer.MAX_VALUE);
94              } else if (long.class == toType || Long.class == toType) {
95                  bigValue = new BigInteger(stringValue);
96                  lowerBound = BigInteger.valueOf(Long.MIN_VALUE);
97                  upperBound = BigInteger.valueOf(Long.MAX_VALUE);
98              } else {
99                  throw new IllegalArgumentException("Unexpected numeric type: " + toType.getName());
100             }
101         } catch (NumberFormatException e) {
102             //shoult it fail here? BigInteger doesnt seem to be so nice parsing numbers as NumberFormat
103             return true;
104         }
105 
106         return ((Comparable)bigValue).compareTo(lowerBound) >= 0 && ((Comparable)bigValue).compareTo(upperBound) <= 0;
107     }
108 
109     private boolean isIntegerType(Class type) {
110         if (double.class == type || float.class == type || Double.class == type || Float.class == type
111                 || char.class == type || Character.class == type) {
112             return false;
113         }
114 
115         return true;
116     }
