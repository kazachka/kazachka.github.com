1   package com.opensymphony.xwork2.conversion;
2   
3   import java.util.Map;
4   
5   /**
6    * Holds all mappings related to {@link TypeConverter}s
7    */
8   public interface TypeConverterHolder {
9   
10      /**
11       * Adds mapping for default type converters - application scoped
12       *
13       * @param className     name of the class with associated converter
14       * @param typeConverter {@link TypeConverter} instance for associated class
15       */
16      void addDefaultMapping(String className, TypeConverter typeConverter);
17  
18      /**
19       * Checks if converter was already defined for given class
20       *
21       * @param className name of the class to check for
22       * @return true if default mapping was already specified
23       */
24      boolean containsDefaultMapping(String className);
25  
26      /**
27       * Returns instance of {@link TypeConverter} associated with given class
28       *
29       * @param className name of the class to return converter for
30       * @return instance of {@link TypeConverter} to be used to convert class
31       */
32      TypeConverter getDefaultMapping(String className);
33  
34      /**
35       * Target class conversion Mappings.
36       *
37       * @param clazz class to convert to/from
38       * @return {@link TypeConverter} for given class
39       */
40      Map<String, Object> getMapping(Class clazz);
41  
42      /**
43       * Assign mapping of converters for given class
44       *
45       * @param clazz   class to convert to/from
46       * @param mapping property converters
47       */
48      void addMapping(Class clazz, Map<String, Object> mapping);
49  
50      /**
51       * Check if there is no mapping for given class to convert
52       *
53       * @param clazz class to convert to/from
54       * @return true if mapping couldn't be found
55       */
56      boolean containsNoMapping(Class clazz);
57  
58      /**
59       * Adds no mapping flag for give class
60       *
61       * @param clazz class to register missing converter
62       */
63      void addNoMapping(Class clazz);
64  
65      /**
66       * Checks if no mapping was defined for given class name
67       * FIXME lukaszlenart: maybe it should be merged with NoMapping
68       *
69       * @param className name of the class to check for
70       * @return true if converter was defined for given class name
71       */
72      boolean containsUnknownMapping(String className);
73  
74      /**
75       * Adds no converter flag for given class name
76       * FIXME lukaszlenart: maybe it should be merged with NoMapping
77       *
78       * @param className name of the class to mark there is no converter for it
79       */
80      void addUnknownMapping(String className);
81  
82  }
