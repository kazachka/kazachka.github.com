3   import org.apache.commons.lang3.StringUtils;
4   
5   import java.lang.reflect.Member;
6   import java.text.DateFormat;
7   import java.text.SimpleDateFormat;
8   import java.util.ArrayList;
9   import java.util.Date;
10  import java.util.List;
11  import java.util.Map;
12  
13  public class StringConverter extends DefaultTypeConverter {
14  
15      @Override
16      public Object convertValue(Map<String, Object> context, Object target, Member member, String propertyName, Object value, Class toType) {
17          String result = null;
18  
19          if (value instanceof int[]) {
20              int[] x = (int[]) value;
21              List<Integer> intArray = new ArrayList<>(x.length);
22  
23              for (int aX : x) {
24                  intArray.add(Integer.valueOf(aX));
25              }
26  
27              result = StringUtils.join(intArray, ", ");
28          } else if (value instanceof long[]) {
29              long[] x = (long[]) value;
30              List<Long> longArray = new ArrayList<>(x.length);
31  
32              for (long aX : x) {
33                  longArray.add(Long.valueOf(aX));
34              }
35  
36              result = StringUtils.join(longArray, ", ");
37          } else if (value instanceof double[]) {
38              double[] x = (double[]) value;
39              List<Double> doubleArray = new ArrayList<>(x.length);
40  
41              for (double aX : x) {
42                  doubleArray.add(new Double(aX));
43              }
44  
45              result = StringUtils.join(doubleArray, ", ");
46          } else if (value instanceof boolean[]) {
47              boolean[] x = (boolean[]) value;
48              List<Boolean> booleanArray = new ArrayList<>(x.length);
49  
50              for (boolean aX : x) {
51                  booleanArray.add(new Boolean(aX));
52              }
53  
54              result = StringUtils.join(booleanArray, ", ");
55          } else if (value instanceof Date) {
56              DateFormat df = null;
57              if (value instanceof java.sql.Time) {
58                  df = DateFormat.getTimeInstance(DateFormat.MEDIUM, getLocale(context));
59              } else if (value instanceof java.sql.Timestamp) {
60                  SimpleDateFormat dfmt = (SimpleDateFormat) DateFormat.getDateTimeInstance(DateFormat.SHORT,
61                          DateFormat.MEDIUM,
62                          getLocale(context));
63                  df = new SimpleDateFormat(dfmt.toPattern() + MILLISECOND_FORMAT);
64              } else {
65                  df = DateFormat.getDateInstance(DateFormat.SHORT, getLocale(context));
66              }
67              result = df.format(value);
68          } else if (value instanceof String[]) {
69              result = StringUtils.join((String[]) value, ", ");
70          }
71          return result;
72      }
73  }
