34   * <p>A timer stack.</p>
35   * <!-- START SNIPPET: profilingAspect_struts2 -->
36   * <p>
37   * Struts2 profiling aspects involves the following:
38   * </p>
39   *
40   * <ul>
41   *  <li>ActionContextCleanUp</li>
42   *  <li>FreemarkerPageFilter</li>
43   *  <li>DispatcherFilter
44   *      <ul>
45   *          <li>Dispatcher
46   *              <ul>
47   *                  <li>creation of DefaultActionProxy
48   *                      <ul>
49   *                          <li>creation of DefaultActionInvocation
50   *                              <ul>
51   *                                  <li>creation of Action</li>
52   *                              </ul>
53   *                          </li>
54   *                      </ul>
55   *                  </li>
56   *                  <li>execution of DefaultActionProxy
57   *                      <ul>
58   *                          <li>invocation of DefaultActionInvocation
59   *                              <ul>
60   *                                  <li>invocation of Interceptors</li>
61   *                                  <li>invocation of Action</li>
62   *                                  <li>invocation of PreResultListener</li>
63   *                                  <li>invocation of Result</li>
64   *                              </ul>
65   *                          </li>
66   *                      </ul>
67   *                  </li>
68   *              </ul>
69   *          </li>
70   *      </ul>
71   *  </li>
72   * </ul>
73   *
74   * <!-- END SNIPPET: profilingAspect_struts2 -->
75   *
76   *
77   * <!-- START SNIPPET: profilingAspect_xwork -->
78   * <p>
79   * XWork2 profiling aspects involves the following:
80   * </p>
81   *
82   * <ul>
83   *  <li>creation of DefaultActionProxy
84   *      <ul>
85   *          <li>creation of DefaultActionInvocation</li>
86   *      </ul>
87   *  </li>
88   *  <li>creation of Action</li>
89   *  <li>execution of DefaultActionProxy
90   *      <ul>
91   *          <li>invocation of DefaultActionInvocation
92   *              <ul>
93   *                  <li>invocation of Interceptors</li>
94   *                  <li>invocation of Action</li>
95   *                  <li>invocation of PreResultListener</li>
96   *                  <li>invocation of Result</li>
97   *              </ul>
98   *          </li>
99   *      </ul>
100  *  </li>
101  * </ul>
102  *
103  * <!-- END SNIPPET: profilingAspect_xwork -->
104  *
105  *
106  * <!-- START SNIPPET: activationDescription -->
107  * <p>
108  * Activating / Deactivating of the profiling feature could be done through:
109  * </p>
110  * <!-- END SNIPPET: activationDescription -->
111  *
112  *
113  *
114  * <p>System properties:</p>
115  * <pre>
116  * <!-- START SNIPPET: activationThroughSystemProperty -->
117  *
118  *  -Dxwork.profile.activate=true
119  *
120  * <!-- END SNIPPET: activationThroughSystemProperty -->
121  * </pre>
122  *
123  * <!-- START SNIPPET: activationThroughSystemPropertyDescription -->
124  * <p>
125  * This could be done in the container startup script eg. CATALINA_OPTS in catalina.sh
126  * (tomcat) or using 'java -Dxwork.profile.activate=true -jar start.jar' (jetty)
127  * </p>
128  * <!-- END SNIPPET: activationThroughSystemPropertyDescription -->
129  *
130  *
131  * <p>Code :</p>
132  * <pre>
133  * <!-- START SNIPPET: activationThroughCode -->
134  *
135  *  UtilTimerStack.setActivate(true);
136  *
137  * <!-- END SNIPPET: activationThroughCode -->
138  * </pre>
139  * <!-- START SNIPPET: activationThroughCodeDescription -->
140  * <p>
141  * This could be done in a static block, in a Spring bean with lazy-init='false',
142  * in a Servlet with init-on-startup as some numeric value, in a Filter or
143  * Listener's init method etc.
144  * </p>
145  * <!-- END SNIPPET: activationThroughCodeDescription -->
146  *
147  * <p>
148  * Parameter:
149  * </p>
150  *
151  * <pre>
152  * <!-- START SNIPPET: activationThroughParameter -->
153  *
154  * &lt;action ... &gt;
155  *  ...
156  *  &lt;interceptor-ref name=&quot;profiling&quot;&gt;
157  *      &lt;param name=&quot;profilingKey&quot;&gt;profiling&lt;/param&gt;
158  *  &lt;/interceptor-ref&gt;
159  *  ...
160  * &lt;/action&gt;
161  *
162  * or
163  *
164  * &lt;action .... &gt;
165  * ...
166  *  &lt;interceptor-ref name=&quot;profiling&quot; /&gt;
167  * ...
168  * &lt;/action&gt;
169  *
170  * through url
171  *
172  * http://host:port/context/namespace/someAction.action?profiling=true
173  *
174  * through code
175  *
176  * ActionContext.getContext().getParameters().put(&quot;profiling&quot;, &quot;true);
177  *
178  * <!-- END SNIPPET: activationThroughParameter -->
179  * </pre>
180  *
181  *
182  * <!-- START SNIPPET: activationThroughParameterDescription -->
183  * <p>
184  * To use profiling activation through parameter, one will need to pass in through
185  * the 'profiling' parameter (which is the default) and could be changed through
186  * the param tag in the interceptor-ref.
187  * </p>
188  * <!-- END SNIPPET: activationThroughParameterDescription -->
189  *
190  * <p>Warning:</p>
191  *
192  * <!-- START SNIPPET: activationThroughParameterWarning -->
193  * <p>
194  * Profiling activation through a parameter requires the following:
195  * </p>
196  *
197  * <ul>
198  * <li>Profiling interceptor in interceptor stack</li>
199  * <li>dev mode on (struts.devMode=true in struts.properties)
200  * </ul>
201  *
202  * <!-- END SNIPPET: activationThroughParameterWarning -->
203  *
204  * <!-- START SNIPPET: filteringDescription -->
205  * <p>
206  * One could filter out the profile logging by having a System property as follows. With this
207  * 'xwork.profile.mintime' property, one could only log profile information when its execution time
208  * exceed those specified in 'xwork.profile.mintime' system property. If no such property is specified,
209  * it will be assumed to be 0, hence all profile information will be logged.
210  * </p>
211  *
212  * <!-- END SNIPPET: filteringDescription -->
213  *
214  * <pre>
215  * <!-- START SNIPPET: filteringCode -->
216  *
217  *  -Dxwork.profile.mintime=10000
218  *
219  * <!-- END SNIPPET: filteringCode -->
220  * </pre>
221  *
222  * <!-- START SNIPPET: methodDescription -->
223  * <p>
224  * One could extend the profiling feature provided by Struts2 in their web application as well.
225  * </p>
226  * <!-- END SNIPPET: methodDescription -->
227  *
228  * <pre>
229  * <!-- START SNIPPET: method1 -->
230  *
231  *    String logMessage = &quot;Log message&quot;;
232  *    UtilTimerStack.push(logMessage);
233  *    try {
234  *        // do some code
235  *    }
236  *    finally {
237  *        UtilTimerStack.pop(logMessage); // this needs to be the same text as above
238  *    }
239  *
240  * <!-- END SNIPPET: method1 -->
241  * </pre>
242  * <p>
243  * or
244  * </p>
245  * <pre>
246  * <!-- START SNIPPET: method2 -->
247  *
248  *   String result = UtilTimerStack.profile(&quot;purchaseItem: &quot;,
249  *       new UtilTimerStack.ProfilingBlock&lt;String&gt;() {
250  *            public String doProfiling() {
251  *               // do some code
252  *               return &quot;Ok&quot;;
253  *            }
254  *       });
255  *
256  * <!-- END SNIPPET: method2 -->
257  * </pre>
258  *
259  *
260  * <!-- START SNIPPET: profileLogFile -->
261  * <p>
262  * Profiled result is logged using commons-logging under the logger named
263  * 'com.opensymphony.xwork2.util.profiling.UtilTimerStack'. Depending on the underlying logging implementation
264  * say if it is Log4j, one could direct the log to appear in a different file, being emailed to someone or have
265  * it stored in the db.
266  * </p>
267  *
268  * <!-- END SNIPPET: profileLogFile -->
269  */
270 public class UtilTimerStack {
271 
272     // A reference to the current ProfilingTimerBean
273     protected static ThreadLocal<ProfilingTimerBean> current = new ThreadLocal<>();
274 
275     /**
276      * System property that controls whether this timer should be used or not.  Set to &quot;true&quot; activates
277      * the timer.  Set to &quot;false&quot; to deactivate.
278      */
279     public static final String ACTIVATE_PROPERTY = "xwork.profile.activate";
280 
281     /**
282      * System property that controls the min time, that if exceeded will cause a log (at INFO level) to be
283      * created.
284      */
285     public static final String MIN_TIME = "xwork.profile.mintime";
286 
287     private static final Logger LOG = LogManager.getLogger(UtilTimerStack.class);
288 
289     /**
290      * Initialized in a static block, it can be changed at runtime by calling setActive(...)
291      */
292     private static boolean active;
293 
294     static {
295         active = "true".equalsIgnoreCase(System.getProperty(ACTIVATE_PROPERTY));
296     }
297 
298     /**
299      * Create and start a performance profiling with the <code>name</code> given. Deal with
300      * profile hierarchy automatically, so caller don't have to be concern about it.
301      *
302      * @param name profile name
303      */
304     public static void push(String name) {
305         if (!isActive()) {
306             return;
307         }
308 
309         //create a new timer and start it
310         ProfilingTimerBean newTimer = new ProfilingTimerBean(name);
311         newTimer.setStartTime();
312 
313         //if there is a current timer - add the new timer as a child of it
314         ProfilingTimerBean currentTimer = (ProfilingTimerBean) current.get();
315         if (currentTimer != null) {
316             currentTimer.addChild(newTimer);
317         }
318 
319         //set the new timer to be the current timer
320         current.set(newTimer);
321     }
322 
323     /**
324      * End a performance profiling with the <code>name</code> given. Deal with
325      * profile hierarchy automatically, so caller don't have to be concern about it.
326      *
327      * @param name profile name
328      */
329     public static void pop(String name) {
330         if (!isActive()) {
331             return;
332         }
333 
334         ProfilingTimerBean currentTimer = current.get();
335 
336         //if the timers are matched up with each other (ie push("a"); pop("a"));
337         if (currentTimer != null && name != null && name.equals(currentTimer.getResource())) {
338             currentTimer.setEndTime();
339             ProfilingTimerBean parent = currentTimer.getParent();
340             //if we are the root timer, then print out the times
341             if (parent == null) {
342                 printTimes(currentTimer);
343                 current.set(null); //for those servers that use thread pooling
344             } else {
345                 current.set(parent);
346             }
347         } else {
348             //if timers are not matched up, then print what we have, and then print warning.
349             if (currentTimer != null) {
350                 printTimes(currentTimer);
351                 current.set(null); //prevent printing multiple times
352                 LOG.warn("Unmatched Timer. Was expecting {}, instead got {}", currentTimer.getResource(), name);
353             }
354         }
355     }
356 
357     /**
358      * Do a log (at INFO level) of the time taken for this particular profiling.
359      *
360      * @param currentTimer profiling timer bean
361      */
362     private static void printTimes(ProfilingTimerBean currentTimer) {
363         LOG.info(currentTimer.getPrintable(getMinTime()));
364     }
365 
366     /**
367      * Get the min time for this profiling, it searches for a System property
368      * 'xwork.profile.mintime' and default to 0.
369      *
370      * @return min time for this profiling
371      */
372     private static long getMinTime() {
373         try {
374             return Long.parseLong(System.getProperty(MIN_TIME, "0"));
375         } catch (NumberFormatException e) {
376             return -1;
377         }
378     }
379 
380     /**
381      * Determine if profiling is being activated, by searching for a system property
382      * 'xwork.profile.activate', default to false (profiling is off).
383      *
384      * @return <tt>true</tt>, if active, <tt>false</tt> otherwise.
385      */
386     public static boolean isActive() {
387         return active;
388     }
389 
390     /**
391      * @param active Turn profiling on or off.
392      */
393     public static void setActive(boolean active) {
394         if (active) {
395             System.setProperty(ACTIVATE_PROPERTY, "true");
396         } else {
397             System.clearProperty(ACTIVATE_PROPERTY);
398         }
399         UtilTimerStack.active = active;
400     }
401 
402 
403     /**
404      * <p>
405      * A convenience method that allows <code>block</code> of code subjected to profiling to be executed
406      * and avoid the need of coding boiler code that does pushing (UtilTimeBean.push(...)) and
407      * poping (UtilTimerBean.pop(...)) in a try ... finally ... block.
408      * </p>
409      *
410      * <p>
411      * Example of usage:
412      * </p>
413      *
414      * <pre>
415      * 	 // we need a returning result
416      *   String result = UtilTimerStack.profile("purchaseItem: ",
417      *       new UtilTimerStack.ProfilingBlock&lt;String&gt;() {
418      *            public String doProfiling() {
419      *               getMyService().purchaseItem(....)
420      *               return "Ok";
421      *            }
422      *       });
423      * </pre>
424      *
425      * or
426      *
427      * <pre>
428      *   // we don't need a returning result
429      *   UtilTimerStack.profile("purchaseItem: ",
430      *       new UtilTimerStack.ProfilingBlock&lt;String&gt;() {
431      *            public String doProfiling() {
432      *               getMyService().purchaseItem(....)
433      *               return null;
434      *            }
435      *       });
436      * </pre>
437      *
438      * @param <T>   any return value if there's one.
439      * @param name  profile name
440      * @param block code block subjected to profiling
441      * @return T
442      * @throws Exception in case of any errors
443      */
444     public static <T> T profile(String name, ProfilingBlock<T> block) throws Exception {
445         UtilTimerStack.push(name);
446         try {
447             return block.doProfiling();
448         } finally {
449             UtilTimerStack.pop(name);
450         }
451     }
452 
453     /**
454      * A callback interface where code subjected to profile is to be executed. This eliminates the need
455      * of coding boiler code that does pushing (UtilTimerBean.push(...)) and poping (UtilTimerBean.pop(...))
456      * in a try ... finally ... block.
457      *
458      * @param <T> type
459      */
460     public static interface ProfilingBlock<T> {
461 
462         /**
463          * Method that execute the code subjected to profiling.
464          *
465          * @return profiles Type
466          * @throws Exception in case of any errors
467          */
468         T doProfiling() throws Exception;
469     }
