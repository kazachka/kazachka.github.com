21  package it.org.apache.struts2.showcase;
22  
23  public class UITagExampleTest extends ITBaseTest {
24      public void testInputForm() {
25          setScriptingEnabled(false);
26          beginAt("/tags/ui/example!input.action");
27          assertFormPresent("exampleSubmit");
28          // text box
29          assertFormElementPresent("name");
30          // textarea
31          assertFormElementPresent("bio");
32          // select
33          assertFormElementPresent("favouriteColor");
34          // checkbox list
35          assertFormElementPresent("friends");
36          // checkbox
37          assertFormElementPresent("legalAge");
38  
39          // set fields
40          setTextField("name", "name");
41          setTextField("bio", "bio");
42          selectOption("favouriteColor", "Red");
43          checkCheckbox("friends", "Patrick");
44          checkCheckbox("friends", "Jason");
45          checkCheckbox("legalAge");
46  
47          submit();
48  
49          assertTextInElement("name", "name");
50          assertTextInElement("bio", "bio");
51          assertTextInElement("favouriteColor", "Red");
52          assertTextInElement("friends", "[Patrick, Jason]");
53          assertTextInElement("legalAge", "true");
54      }
55  }
