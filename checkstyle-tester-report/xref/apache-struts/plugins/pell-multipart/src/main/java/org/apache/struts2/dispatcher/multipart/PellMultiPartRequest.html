42  /**
43   * Multipart form data request adapter for Jason Pell's multipart utils package.
44   *
45   */
46  public class PellMultiPartRequest implements MultiPartRequest {
47  
48      private static final Logger LOG = LogManager.getLogger(PellMultiPartRequest.class);
49      private ServletMultipartRequest multi;
50  
51      private String defaultEncoding;
52      private boolean maxSizeProvided;
53      private int maxSize;
54      
55      @Inject(StrutsConstants.STRUTS_I18N_ENCODING)
56      public void setDefaultEncoding(String enc) {
57          this.defaultEncoding = enc;
58      }
59      
60      @Inject(StrutsConstants.STRUTS_MULTIPART_MAXSIZE)
61      public void setMaxSize(String maxSize) {
62      	this.maxSizeProvided = true;
63          this.maxSize = Integer.parseInt(maxSize);
64      }
65      
66      /**
67       * Creates a new request wrapper to handle multi-part data using methods adapted from Jason Pell's
68       * multipart classes (see class description).
69       *
70       * @param saveDir        the directory to save off the file
71       * @param servletRequest the request containing the multipart
72       */
73      public void parse(HttpServletRequest servletRequest, String saveDir) throws IOException {
74          //this needs to be synchronised, as we should not change the encoding at the same time as
75          //calling the constructor.  See javadoc for MultipartRequest.setEncoding().
76          synchronized (this) {
77              setEncoding();
78              if (maxSizeProvided){
79              	multi = new ServletMultipartRequest(servletRequest, saveDir, maxSize);
80              }else{
81              	multi = new ServletMultipartRequest(servletRequest, saveDir);
82              }
83          }
84      }
85      
86      public Enumeration getFileParameterNames() {
87          return multi.getFileParameterNames();
88      }
89  
90      public String[] getContentType(String fieldName) {
91          return new String[]{multi.getContentType(fieldName)};
92      }
93  
94      public File[] getFile(String fieldName) {
95          return new File[]{multi.getFile(fieldName)};
96      }
97  
98      public String[] getFileNames(String fieldName) {
99  
100         // TODO - not sure about this - is this the filename of the actual file or
101         // TODO - the uploaded filename as provided by the browser?
102         // TODO - Not sure what version of Pell this class uses as it doesn't seem to be the latest
103         return new String[]{multi.getFile(fieldName).getName()};
104     }
105 
106     public String[] getFilesystemName(String fieldName) {
107         return new String[]{multi.getFileSystemName(fieldName)};
108     }
109 
110     public String getParameter(String name) {
111         return multi.getURLParameter(name);
112     }
113 
114     public Enumeration<String> getParameterNames() {
115         return multi.getParameterNames();
116     }
117 
118     public String[] getParameterValues(String name) {
119         Enumeration enumeration = multi.getURLParameters(name);
120 
121         if (!enumeration.hasMoreElements()) {
122             return null;
123         }
124 
125         List<String> values = new ArrayList<String>();
126 
127         while (enumeration.hasMoreElements()) {
128             values.add((String) enumeration.nextElement());
129         }
130 
131         return values.toArray(new String[values.size()]);
132     }
133 
134     public List<String> getErrors() {
135         return Collections.emptyList();
136     }
137 
138     /**
139      * Sets the encoding for the uploaded params.  This needs to be set if you are using character sets other than
140      * ASCII.
141      * <p>
142      * The encoding is looked up from the configuration setting 'struts.i18n.encoding'.  This is usually set in
143      * default.properties & struts.properties.
144      * </p>
145      */
146     private void setEncoding() {
147         String encoding = null;
148 
149         try {
150             encoding = defaultEncoding;
151 
152             if (encoding != null) {
153                 //NB: This should never be called at the same time as the constructor for
154                 //ServletMultiPartRequest, as it can cause problems.
155                 //See javadoc for MultipartRequest.setEncoding()
156                 http.utils.multipartrequest.MultipartRequest.setEncoding(encoding);
157             } else {
158                 http.utils.multipartrequest.MultipartRequest.setEncoding("UTF-8");
159             }
160         } catch (IllegalArgumentException e) {
161             if (LOG.isInfoEnabled()) {
162         	    LOG.info("Could not get encoding property 'struts.i18n.encoding' for file upload.  Using system default");
163             }
164         } catch (UnsupportedEncodingException e) {
165             LOG.error("Encoding " + encoding + " is not a valid encoding.  Please check your struts.properties file.");
166         }
167     }
168 
169     /* (non-Javadoc)
170     * @see org.apache.struts2.dispatcher.multipart.MultiPartRequest#cleanUp()
171     */
172     public void cleanUp() {
173         Enumeration fileParameterNames = multi.getFileParameterNames();
174         while (fileParameterNames != null && fileParameterNames.hasMoreElements()) {
175             String inputValue = (String) fileParameterNames.nextElement();
176             File[] files = getFile(inputValue);
177             for (File currentFile : files) {
178                 if (LOG.isInfoEnabled()) {
179                     String msg = LocalizedTextUtil.findText(this.getClass(), "struts.messages.removing.file", Locale.ENGLISH,
180                             "no.message.found", new Object[]{inputValue, currentFile});
181                     LOG.info(msg);
182                 }
183                 if ((currentFile != null) && currentFile.isFile()) {
184                     if (!currentFile.delete()) {
185                         LOG.warn("Resource Leaking: Could not remove uploaded file [{}]", currentFile.getAbsolutePath());
186                     }
187                 }
188             }
189         }
190     }
