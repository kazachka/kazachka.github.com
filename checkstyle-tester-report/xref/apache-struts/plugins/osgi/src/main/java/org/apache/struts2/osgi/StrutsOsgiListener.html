1   package org.apache.struts2.osgi;
2   
3   import org.apache.logging.log4j.Logger;
4   import org.apache.logging.log4j.LogManager;
5   import org.apache.struts2.StrutsException;
6   import org.apache.struts2.osgi.host.OsgiHost;
7   
8   import javax.servlet.ServletContext;
9   import javax.servlet.ServletContextEvent;
10  import javax.servlet.ServletContextListener;
11  
12  /**
13   * ServletContextListener that starts Osgi host
14   */
15  public class StrutsOsgiListener implements ServletContextListener {
16  
17      public static final String OSGI_HOST = "__struts_osgi_host";
18      public static final String PLATFORM_KEY = "struts.osgi.host";
19  
20      private static final Logger LOG = LogManager.getLogger(StrutsOsgiListener.class);
21  
22      private OsgiHost osgiHost;
23  
24      public void contextInitialized(ServletContextEvent sce) {
25          ServletContext servletContext = sce.getServletContext();
26  
27          String platform = servletContext.getInitParameter(PLATFORM_KEY);
28          LOG.debug("Defined OSGi platform as [{}] via context-param [{}]", platform, PLATFORM_KEY);
29  
30          osgiHost = OsgiHostFactory.createOsgiHost(platform);
31          servletContext.setAttribute(OSGI_HOST, osgiHost);
32          try {
33              osgiHost.init(servletContext);
34          } catch (Exception e) {
35              throw new StrutsException("Cannot init OSGi platform!", e);
36          }
37      }
38  
39      public void contextDestroyed(ServletContextEvent sce) {
40          try {
41              osgiHost.destroy();
42          } catch (Exception e) {
43              throw new StrutsException("Cannot stop OSGi platform!", e);
44          }
45      }
46  
47  }
