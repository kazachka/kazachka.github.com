4   package com.opensymphony.xwork2.config.impl;
5   
6   import com.opensymphony.xwork2.inject.Context;
7   import com.opensymphony.xwork2.inject.Factory;
8   import com.opensymphony.xwork2.util.location.Located;
9   import com.opensymphony.xwork2.util.location.LocationUtils;
10  
11  /**
12   * Factory that remembers where a constant came from
13   */
14  public class LocatableConstantFactory<T> extends Located implements Factory {
15      T constant;
16      public LocatableConstantFactory(T constant, Object location) {
17          this.constant = constant;
18          setLocation(LocationUtils.getLocation(location));
19      }
20      
21      public T create(Context ignored) {
22          return constant;
23      }
24      
25      @Override
26      public String toString() {
27          StringBuilder sb = new StringBuilder();
28          sb.append(super.toString());
29          sb.append(" defined at ");
30          sb.append(getLocation().toString());
31          return sb.toString();
32      }
33  
34  }
