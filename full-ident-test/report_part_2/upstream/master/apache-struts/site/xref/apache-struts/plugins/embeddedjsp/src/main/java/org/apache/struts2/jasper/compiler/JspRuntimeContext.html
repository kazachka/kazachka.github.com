44  /**
45   * <p>
46   * Class for tracking JSP compile time file dependencies when the
47   * &#060;%{@literal @}include file="..."%&#062; directive is used.
48   * </p>
49   *
50   * <p>
51   * A background thread periodically checks the files a JSP page
52   * is dependent upon.  If a dependent file changes the JSP page
53   * which included it is recompiled.
54   * </p>
55   *
56   * <p>
57   * Only used if a web application context is a directory.
58   * </p>
59   *
60   * @author Glenn L. Nielsen
61   * @version $Revision: 505593 $
62   */
63  public final class JspRuntimeContext {
64  
65      // Logger
66      private Log log = LogFactory.getLog(JspRuntimeContext.class);
67  
68      /*
69       * Counts how many times the webapp's JSPs have been reloaded.
70       */
71      private int jspReloadCount;
72  
73      /**
74       * Preload classes required at runtime by a JSP servlet so that
75       * we don't get a defineClassInPackage security exception.
76       */
77      static {
78          JspFactoryImpl factory = new JspFactoryImpl();
79          SecurityClassLoad.securityClassLoad(factory.getClass().getClassLoader());
80          if (System.getSecurityManager() != null) {
81              String basePackage = "org.apache.struts2.jasper.";
82              try {
83                  factory.getClass().getClassLoader().loadClass(basePackage +
84                          "runtime.JspFactoryImpl$PrivilegedGetPageContext");
85                  factory.getClass().getClassLoader().loadClass(basePackage +
86                          "runtime.JspFactoryImpl$PrivilegedReleasePageContext");
87                  factory.getClass().getClassLoader().loadClass(basePackage +
88                          "runtime.JspRuntimeLibrary");
89                  factory.getClass().getClassLoader().loadClass(basePackage +
90                          "runtime.JspRuntimeLibrary$PrivilegedIntrospectHelper");
91                  factory.getClass().getClassLoader().loadClass(basePackage +
92                          "runtime.ServletResponseWrapperInclude");
93                  factory.getClass().getClassLoader().loadClass(basePackage +
94                          "servlet.JspServletWrapper");
95              } catch (ClassNotFoundException ex) {
96                  throw new IllegalStateException(ex);
97              }
98          }
99  
100         JspFactory.setDefaultFactory(factory);
101     }

105     /**
106      * <p>
107      * Create a JspRuntimeContext for a web application context.
108      * </p>
109      *
110      * <p>
111      * Loads in any previously generated dependencies from file.
112      * </p>
113      *
114      * @param context ServletContext for web application
115      * @param options options
116      */
117     public JspRuntimeContext(ServletContext context, Options options) {
118 
119         this.context = context;
120         this.options = options;
121 
122         // Get the parent class loader
123         parentClassLoader =
124                 (URLClassLoader) Thread.currentThread().getContextClassLoader();
125         if (parentClassLoader == null) {
126             parentClassLoader =
127                     (URLClassLoader) this.getClass().getClassLoader();
128         }
129 
130         if (log.isDebugEnabled()) {
131             if (parentClassLoader != null) {
132                 log.debug(Localizer.getMessage("jsp.message.parent_class_loader_is",
133                         parentClassLoader.toString()));
134             } else {
135                 log.debug(Localizer.getMessage("jsp.message.parent_class_loader_is",
136                         "<none>"));
137             }
138         }
139 
140         initClassPath();
141 
142         if (context instanceof JspCServletContext) {
143             return;
144         }
145 
146         if (Constants.IS_SECURITY_ENABLED) {
147             initSecurity();
148         }
149 
150         // If this web application context is running from a
151         // directory, start the background compilation thread
152         String appBase = context.getRealPath("/");
153         if (!options.getDevelopment()
154                 && appBase != null
155                 && options.getCheckInterval() > 0) {
156             lastCheck = System.currentTimeMillis();
157         }
158     }

284     /**
285      * Method used by background thread to check the JSP dependencies
286      * registered with this class for JSP's.
287      */
288     public void checkCompile() {
289 
290         if (lastCheck < 0) {
291             // Checking was disabled
292             return;
293         }
294         long now = System.currentTimeMillis();
295         if (now > (lastCheck + (options.getCheckInterval() * 1000L))) {
296             lastCheck = now;
297         } else {
298             return;
299         }
300 
301         Object[] wrappers = jsps.values().toArray();
302         for (Object wrapper : wrappers) {
303             JspServletWrapper jsw = (JspServletWrapper) wrapper;
304             JspCompilationContext ctxt = jsw.getJspEngineContext();
305             // JspServletWrapper also synchronizes on this when
306             // it detects it has to do a reload
307             synchronized (jsw) {
308                 try {
309                     ctxt.compile();
310                 } catch (FileNotFoundException ex) {
311                     ctxt.incrementRemoved();
312                 } catch (Throwable t) {
313                     jsw.getServletContext().log("Background compile failed",
314                             t);
315                 }
316             }
317         }
318 
319     }

332     /**
333      * Method used to initialize classpath for compiles.
334      */
335     private void initClassPath() {
336 
337         URL[] urls = parentClassLoader.getURLs();
338         StringBuffer cpath = new StringBuffer();
339         String sep = System.getProperty("path.separator");
340 
341         for (int i = 0; i < urls.length; i++) {
342             // Tomcat 4 can use URL's other than file URL's,
343             // a protocol other than file: will generate a
344             // bad file system path, so only add file:
345             // protocol URL's to the classpath.
346 
347             if (urls[i].getProtocol().equals("file")) {
348                 cpath.append((String) urls[i].getFile() + sep);
349             }
350         }
351 
352         cpath.append(options.getScratchDir() + sep);
353 
354         String cp = (String) context.getAttribute(Constants.SERVLET_CLASSPATH);
355         if (cp == null || cp.equals("")) {
356             cp = options.getClassPath();
357         }
358 
359         classpath = cpath.toString() + cp;
360 
361         if (log.isDebugEnabled()) {
362             log.debug("Compilation classpath initialized: " + getClassPath());
363         }
364     }

366     /**
367      * Method used to initialize SecurityManager data.
368      */
369     private void initSecurity() {
370 
371         // Setup the PermissionCollection for this web app context
372         // based on the permissions configured for the root of the
373         // web app context directory, then add a file read permission
374         // for that directory.
375         Policy policy = Policy.getPolicy();
376         if (policy != null) {
377             try {
378                 // Get the permissions for the web app context
379                 String docBase = context.getRealPath("/");
380                 if (docBase == null) {
381                     docBase = options.getScratchDir().toString();
382                 }
383                 String codeBase = docBase;
384                 if (!codeBase.endsWith(File.separator)) {
385                     codeBase = codeBase + File.separator;
386                 }
387                 File contextDir = new File(codeBase);
388                 URL url = contextDir.getCanonicalFile().toURL();
389                 codeSource = new CodeSource(url, (Certificate[]) null);
390                 permissionCollection = policy.getPermissions(codeSource);
391 
392                 // Create a file read permission for web app context directory
393                 if (!docBase.endsWith(File.separator)) {
394                     permissionCollection.add
395                             (new FilePermission(docBase, "read"));
396                     docBase = docBase + File.separator;
397                 } else {
398                     permissionCollection.add
399                             (new FilePermission
400                                     (docBase.substring(0, docBase.length() - 1), "read"));
401                 }
402                 docBase = docBase + "-";
403                 permissionCollection.add(new FilePermission(docBase, "read"));
404 
405                 // Create a file read permission for web app tempdir (work)
406                 // directory
407                 String workDir = options.getScratchDir().toString();
408                 if (!workDir.endsWith(File.separator)) {
409                     permissionCollection.add
410                             (new FilePermission(workDir, "read"));
411                     workDir = workDir + File.separator;
412                 }
413                 workDir = workDir + "-";
414                 permissionCollection.add(new FilePermission(workDir, "read"));
415 
416                 // Allow the JSP to access org.apache.struts2.jasper.runtime.HttpJspBase
417                 permissionCollection.add(new RuntimePermission(
418                         "accessClassInPackage.org.apache.struts2.jasper.runtime"));
419 
420                 if (parentClassLoader instanceof URLClassLoader) {
421                     URL[] urls = parentClassLoader.getURLs();
422                     String jarUrl = null;
423                     String jndiUrl = null;
424                     for (int i = 0; i < urls.length; i++) {
425                         if (jndiUrl == null
426                                 && urls[i].toString().startsWith("jndi:")) {
427                             jndiUrl = urls[i].toString() + "-";
428                         }
429                         if (jarUrl == null
430                                 && urls[i].toString().startsWith("jar:jndi:")
431                                 ) {
432                             jarUrl = urls[i].toString();
433                             jarUrl = jarUrl.substring(0, jarUrl.length() - 2);
434                             jarUrl = jarUrl.substring(0,
435                                     jarUrl.lastIndexOf('/')) + "/-";
436                         }
437                     }
438                     if (jarUrl != null) {
439                         permissionCollection.add(
440                                 new FilePermission(jarUrl, "read"));
441                         permissionCollection.add(
442                                 new FilePermission(jarUrl.substring(4), "read"));
443                     }
444                     if (jndiUrl != null)
445                         permissionCollection.add(
446                                 new FilePermission(jndiUrl, "read"));
447                 }
448             } catch (Exception e) {
449                 context.log("Security Init for context failed", e);
450             }
451         }
452     }
