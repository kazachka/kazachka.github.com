282     public void testSkipUserMarkerActionLevelShortCircuit() {
283         // get validators
284         List validatorList = annotationActionValidatorManager.getValidators(AnnotationUser.class, null);
285         assertEquals(10, validatorList.size());
286 
287         try {
288             AnnotationUser user = new AnnotationUser();
289             user.setName("Mark");
290             user.setEmail("bad_email");
291             user.setEmail2("bad_email");
292 
293             ValidatorContext context = new GenericValidatorContext(user);
294             annotationActionValidatorManager.validate(user, null, context);
295             assertTrue(context.hasFieldErrors());
296 
297             // check field errors
298             List<String> l = context.getFieldErrors().get("email");
299             assertNotNull(l);
300             assertEquals(1, l.size());
301             assertEquals("Not a valid e-mail.", l.get(0));
302             l = context.getFieldErrors().get("email2");
303             assertNotNull(l);
304             assertEquals(2, l.size());
305             assertEquals("Not a valid e-mail2.", l.get(0));
306             assertEquals("Email2 not from the right company.", l.get(1));
307 
308             // check action errors
309             assertTrue(context.hasActionErrors());
310             l = (List<String>) context.getActionErrors();
311             assertNotNull(l);
312             assertEquals(2, l.size()); // both expression test failed see AnnotationUser-validation.xml
313             assertEquals("Email does not start with mark", l.get(0));
314         } catch (ValidationException ex) {
315             ex.printStackTrace();
316             fail("Validation error: " + ex.getMessage());
317         }
318     }

321         // get validators
322         List validatorList = annotationActionValidatorManager.getValidators(AnnotationUser.class, null);
323         assertEquals(10, validatorList.size());
324 
325         try {
326             AnnotationUser user = new AnnotationUser();
327             user.setName("Mark");
328             // * mark both email to starts with mark to get pass the action-level validator,
329             // so we could concentrate on testing the field-level validators (AnnotationUser-validation.xml)
330             // * make both email the same to pass the action-level validator at 
331             // AnnotationUserMarker-validation.xml
332             user.setEmail("mark_bad_email_for_field_val@foo.com");
333             user.setEmail2("mark_bad_email_for_field_val@foo.com");
334 
335             ValidatorContext context = new GenericValidatorContext(user);
336             annotationActionValidatorManager.validate(user, null, context);
337             assertTrue(context.hasFieldErrors());
338 
339             // check field errors
340             // we have an error in this field level, email does not ends with mycompany.com
341             List l = (List) context.getFieldErrors().get("email");
342             assertNotNull(l);
343             assertEquals(1, l.size()); // because email-field-val is short-circuit
344             assertEquals("Email not from the right company.", l.get(0));
345 
346             
347             // check action errors
348             l = (List) context.getActionErrors();
349             assertFalse(context.hasActionErrors());
350             assertEquals(0, l.size());
351             
352             
353         } catch (ValidationException ex) {

360     public void testActionLevelShortCircuit() throws Exception {
361     	
362     	List validatorList = annotationActionValidatorManager.getValidators(AnnotationUser.class, null);
363         assertEquals(10, validatorList.size());
364         
365         AnnotationUser user = new AnnotationUser();
366         // all fields will trigger error, but sc of action-level, cause it to not appear
367         user.setName(null);		
368 
369         user.setEmail("rainerh(at)example.com");
370         user.setEmail("rainer_h(at)example.com");
371 
372 
373         ValidatorContext context = new GenericValidatorContext(user);
374         annotationActionValidatorManager.validate(user, null, context);
375     	
376     	// check field level errors
377         // shouldn't have any because action error prevents validation of anything else
378         List l = (List) context.getFieldErrors().get("email2");
379         assertNull(l);
380     	
381     	
382         // check action errors
383         assertTrue(context.hasActionErrors());
384         l = (List) context.getActionErrors();
385         assertNotNull(l);
386         // we only get one, because AnnotationUserMarker-validation.xml action-level validator
387         // already sc it   :-)
388         assertEquals(1, l.size()); 
389         assertEquals("Email not the same as email2", l.get(0));
390     }
