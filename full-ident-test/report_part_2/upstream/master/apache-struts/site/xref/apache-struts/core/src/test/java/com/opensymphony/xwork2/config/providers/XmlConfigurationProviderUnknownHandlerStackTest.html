13      public void testStackWithElements() throws ConfigurationException {

15          ConfigurationProvider provider = buildConfigurationProvider(filename);
16          loadConfigurationProviders(provider);
17          configurationManager.reload();
18  
19          List<UnknownHandlerConfig> unknownHandlerStack = configuration.getUnknownHandlerStack();

25  
26          UnknownHandlerManager unknownHandlerManager = new DefaultUnknownHandlerManager();
