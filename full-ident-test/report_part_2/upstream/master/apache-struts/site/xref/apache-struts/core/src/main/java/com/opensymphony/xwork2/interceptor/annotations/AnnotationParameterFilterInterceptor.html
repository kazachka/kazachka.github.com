18  /**
19   * Annotation based version of {@link ParameterFilterInterceptor}.
20   *
21   * <p>
22   * This {@link Interceptor} must be placed in the stack before the {@link ParametersInterceptor}
23   * When a parameter matches a field that is marked {@link Blocked} then it is removed from
24   * the parameter map.
25   * </p>
26   * <p>
27   * If an {@link Action} class is marked with {@link BlockByDefault} then all parameters are
28   * removed unless a field on the Action exists and is marked with {@link Allowed}
29   * </p>
30   *
31   * @author martin.gilday
32   */
33  public class AnnotationParameterFilterInterceptor extends AbstractInterceptor {
34  
35      /* (non-Javadoc)
36        * @see com.opensymphony.xwork2.interceptor.AbstractInterceptor#intercept(com.opensymphony.xwork2.ActionInvocation)
37        */
38      @Override public String intercept(ActionInvocation invocation) throws Exception {
39  
40          final Object action = invocation.getAction();
41          HttpParameters parameters = invocation.getInvocationContext().getParameters();
42  
43          Object model = invocation.getStack().peek();
44          if (model == action) {
45              model = null;
46          }
47  
48          boolean blockByDefault = action.getClass().isAnnotationPresent(BlockByDefault.class);
49          List<Field> annotatedFields = new ArrayList<>();
50  
51          if (blockByDefault) {
52              AnnotationUtils.addAllFields(Allowed.class, action.getClass(), annotatedFields);
53              if (model != null) {
54                  AnnotationUtils.addAllFields(Allowed.class, model.getClass(), annotatedFields);
55              }
56  
57              for (String paramName : parameters.getNames()) {
58                  boolean allowed = false;
59  
60                  for (Field field : annotatedFields) {
61                      //TODO only matches exact field names.  need to change to it matches start of ognl expression
62                      //i.e take param name up to first . (period) and match against that
63                      if (field.getName().equals(paramName)) {
64                          allowed = true;
65                          break;
66                      }
67                  }
68  
69                  if (!allowed) {
70                      parameters = parameters.remove(paramName);
71                  }
72              }
73          } else {
74              AnnotationUtils.addAllFields(Blocked.class, action.getClass(), annotatedFields);
75              if (model != null) {
76                  AnnotationUtils.addAllFields(Blocked.class, model.getClass(), annotatedFields);
77              }
78  
79              for (String paramName : parameters.getNames()) {
80                  for (Field field : annotatedFields) {
81                      //TODO only matches exact field names.  need to change to it matches start of ognl expression
82                      //i.e take param name up to first . (period) and match against that
83                      if (field.getName().equals(paramName)) {
84                          parameters = parameters.remove(paramName);
85                      }
86                  }
87              }
88          }
89  
90          invocation.getInvocationContext().setParameters(parameters);
91  
92          return invocation.invoke();
93      }
94  
95  }
