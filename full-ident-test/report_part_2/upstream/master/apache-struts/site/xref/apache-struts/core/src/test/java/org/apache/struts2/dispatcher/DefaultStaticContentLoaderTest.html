9       public void testParsePackages() throws Exception {

11          DefaultStaticContentLoader filterDispatcher = new DefaultStaticContentLoader();
12          List<String> result1 = filterDispatcher.parse("foo.bar.package1 foo.bar.package2 foo.bar.package3");
13          List<String> result2 = filterDispatcher.parse("foo.bar.package1\tfoo.bar.package2\tfoo.bar.package3");
14          List<String> result3 = filterDispatcher.parse("foo.bar.package1,foo.bar.package2,foo.bar.package3");
15          List<String> result4 = filterDispatcher.parse("foo.bar.package1    foo.bar.package2  \t foo.bar.package3   , foo.bar.package4");
16  
17          assertEquals(result1.get(0), "foo/bar/package1/");
18          assertEquals(result1.get(1), "foo/bar/package2/");
19          assertEquals(result1.get(2), "foo/bar/package3/");
20  
21          assertEquals(result2.get(0), "foo/bar/package1/");
22          assertEquals(result2.get(1), "foo/bar/package2/");
23          assertEquals(result2.get(2), "foo/bar/package3/");
24  
25          assertEquals(result3.get(0), "foo/bar/package1/");
26          assertEquals(result3.get(1), "foo/bar/package2/");
27          assertEquals(result3.get(2), "foo/bar/package3/");
28  
29          assertEquals(result4.get(0), "foo/bar/package1/");
30          assertEquals(result4.get(1), "foo/bar/package2/");
