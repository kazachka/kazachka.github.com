18  /**
19   * Base class used to test action in portlet environment
20   */
21  public abstract class StrutsPortletTestCase extends StrutsTestCase {
22  
23      private static final Logger LOG = LogManager.getLogger(StrutsPortletTestCase.class);
24  
25      protected MockPortletSession portletSession;
26      protected MockPortletRequest portletRequest;
27      protected MockPortletResponse portletResponse;
28      protected MockContext portletContext;
29  
30      @Override
31      protected void initActionContext(ActionContext actionContext) {
32          super.initActionContext(actionContext);
33          initPortletContext(actionContext);
34      }
35  
36      protected void initPortletContext(ActionContext actionContext) {
37          LOG.debug("Initializing mock portlet environment");
38          portletContext = new MockContext();
39          portletContext.setMajorVersion(getMajorVersion());
40          actionContext.put(StrutsStatics.STRUTS_PORTLET_CONTEXT, portletContext);
41  
42          portletRequest = new MockPortletRequest(portletContext);
43          portletResponse = new MockStateAwareResponse();
44          portletSession = new MockPortletSession();
45          portletRequest.setSession(portletSession);
46          actionContext.setSession(createSession());
47          actionContext.put(PortletConstants.REQUEST, portletRequest);
48          actionContext.put(PortletConstants.RESPONSE, portletResponse);
49          actionContext.put(PortletConstants.MODE_NAMESPACE_MAP, new HashMap<PortletMode, String>());
50          actionContext.put(PortletConstants.PHASE, PortletPhase.EVENT_PHASE);
51      }
52  
53      /**
54       * Override to define version of your portlet environment
55       *
56       * @return portlet version
57       */
58      protected int getMajorVersion() {
59          return 2;
60      }
61  
62      /**
63       * Override to create your own session
64       *
65       * @return Map with session parameters
66       */
67      private Map<String, Object> createSession() {
68          return new HashMap<String, Object>(portletRequest.getPortletSession().getAttributeMap());
69      }
70  
71  }
72

76  class MockContext extends MockPortletContext {
77  
78      private int majorVersion;
79  
80      @Override
81      public int getMajorVersion() {
82          return majorVersion;
83      }
84

86          this.majorVersion = majorVersion;
87      }
88  
89  }
