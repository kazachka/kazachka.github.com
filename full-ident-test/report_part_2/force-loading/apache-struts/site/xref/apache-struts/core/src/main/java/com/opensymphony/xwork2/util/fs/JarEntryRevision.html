3   import com.opensymphony.xwork2.FileManager;
4   import org.apache.logging.log4j.Logger;
5   import org.apache.logging.log4j.LogManager;
6   import org.apache.commons.io.FileUtils;
7   
8   import java.io.IOException;
9   import java.net.URL;
10  import java.util.jar.JarFile;
11  import java.util.zip.ZipEntry;
12  
13  /**
14   * Represents jar resource revision, used for jar://* resource
15   */
16  public class JarEntryRevision extends Revision {
17  
18      private static Logger LOG = LogManager.getLogger(JarEntryRevision.class);
19  
20      private static final String JAR_FILE_NAME_SEPARATOR = "!/";
21      private static final String JAR_FILE_EXTENSION_END = ".jar/";
22  
23      private String jarFileName;
24      private String fileNameInJar;
25      private long lastModified;
26  
27      public static Revision build(URL fileUrl, FileManager fileManager) {
28          // File within a Jar
29          // Find separator index of jar filename and filename within jar
30          String jarFileName = "";
31          try {
32              String fileName = fileUrl.toString();
33              int separatorIndex = fileName.indexOf(JAR_FILE_NAME_SEPARATOR);
34              if (separatorIndex == -1) {
35                  separatorIndex = fileName.lastIndexOf(JAR_FILE_EXTENSION_END);
36              }
37              if (separatorIndex == -1) {
38                  LOG.warn("Could not find end of jar file!");
39                  return null;
40              }
41  
42              // Split file name
43              jarFileName = fileName.substring(0, separatorIndex);
44              int index = separatorIndex + JAR_FILE_NAME_SEPARATOR.length();
45              String fileNameInJar = fileName.substring(index).replaceAll("%20", " ");
46  
47              URL url = fileManager.normalizeToFileProtocol(fileUrl);
48              if (url != null) {
49                  JarFile jarFile = new JarFile(FileUtils.toFile(url));
50                  ZipEntry entry = jarFile.getEntry(fileNameInJar);
51                  return new JarEntryRevision(jarFileName, fileNameInJar, entry.getTime());
52              } else {
53                  return null;
54              }
55          } catch (Throwable e) {
56              LOG.warn("Could not create JarEntryRevision for [{}]!", jarFileName, e);
57              return null;
58          }
59      }
60  
61      private JarEntryRevision(String jarFileName, String fileNameInJar, long lastModified) {
62          if ((jarFileName == null) || (fileNameInJar == null)) {
63              throw new IllegalArgumentException("JarFileName and FileNameInJar cannot be null");
64          }
65          this.jarFileName = jarFileName;
66          this.fileNameInJar = fileNameInJar;
67          this.lastModified = lastModified;
68      }
69  
70      public boolean needsReloading() {
71          ZipEntry entry;
72          try {
73              JarFile jarFile = new JarFile(this.jarFileName);
74              entry = jarFile.getEntry(this.fileNameInJar);
75          } catch (IOException e) {
76              entry = null;
77          }
78  
79          return entry != null && (lastModified < entry.getTime());
80      }
81  
82  }
