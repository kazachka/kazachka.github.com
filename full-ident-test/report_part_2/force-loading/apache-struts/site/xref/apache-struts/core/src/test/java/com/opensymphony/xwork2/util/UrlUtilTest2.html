16  public class UrlUtilTest2 extends TestCase {
17  
18      public void testOpenWithJarProtocol() throws IOException {
19          FileManager fileManager = new DefaultFileManager();
20  
21          URL url = ClassLoaderUtil.getResource("xwork-jar.jar", ClassLoaderUtil.class);
22          URL jarUrl = new URL("jar", "", url.toExternalForm() + "!/");
23          URL outputURL = fileManager.normalizeToFileProtocol(jarUrl);
24  
25          assertNotNull(outputURL);
26          assertUrlCanBeOpened(outputURL);
27      }
28  
29      private void assertUrlCanBeOpened(URL url) throws IOException {
30          InputStream is = url.openStream();
31          try (JarInputStream jarStream = new JarInputStream(is)) {
32              assertNotNull(jarStream);
33          }
34      }
