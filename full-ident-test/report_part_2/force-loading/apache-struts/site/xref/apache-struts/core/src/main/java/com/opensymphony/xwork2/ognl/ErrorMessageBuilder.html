3   /**
4    * Helper class to build error messages.
5    */
6   public class ErrorMessageBuilder {
7   
8       private StringBuilder message = new StringBuilder();
9   
10      public static ErrorMessageBuilder create() {
11          return new ErrorMessageBuilder();
12      }
13  
14      private ErrorMessageBuilder() {
15      }
16  
17      public ErrorMessageBuilder errorSettingExpressionWithValue(String expr, Object value) {
18          appenExpression(expr);
19          if (value instanceof Object[]) {
20              appendValueAsArray((Object[]) value, message);
21          } else {
22              appendValue(value);
23          }
24          return this;
25      }
26  
27      private void appenExpression(String expr) {
28          message.append("Error setting expression '");
29          message.append(expr);
30          message.append("' with value ");
31      }
32  
33      private void appendValue(Object value) {
34          message.append("'");
35          message.append(value);
36          message.append("'");
37      }
38  
39      private void appendValueAsArray(Object[] valueArray, StringBuilder msg) {
40          msg.append("[");
41          for (int index = 0; index < valueArray.length; index++) {
42              appendValue(valueArray[index]);
43              if (hasMoreElements(valueArray, index)) {
44                  msg.append(", ");
45              }
46          }
47          msg.append("]");
48      }
49  
50      private boolean hasMoreElements(Object[] valueArray, int index) {
51          return index < (valueArray.length + 1);
52      }
53  
54      public String build() {
55          return message.toString();
56      }
57  
58  }
