26      public void testSessionWasInvalidated() throws Exception {
27          // given
28          ActionContext actionContext = new ActionContext(new HashMap());
29          actionContext.put(ActionContext.PARAMETERS, new LinkedHashMap());
30  
31          ActionInvocation mockActionInvocation = EasyMock.createControl().createMock(ActionInvocation.class);
32  
33          mockActionInvocation.getInvocationContext();
34          EasyMock.expectLastCall().andReturn(actionContext);
35          EasyMock.expectLastCall().anyTimes();
36  
37          EasyMock.replay(mockActionInvocation);
38  
39          HttpServletRequest mockedRequest = EasyMock.createControl().createMock(HttpServletRequest.class);
40          mockedRequest.getSession(false);
41          EasyMock.expectLastCall().andReturn(null);
42          EasyMock.expectLastCall().once();
43          ServletActionContext.setRequest(mockedRequest);
44  
45          EasyMock.replay(mockedRequest);
46  
47          HttpServletResponse mockedResponse = EasyMock.createControl().createMock(HttpServletResponse.class);
48          mockedResponse.isCommitted();
49          EasyMock.expectLastCall().andReturn(false);
50          EasyMock.expectLastCall().once();
51          ServletActionContext.setResponse(mockedResponse);
52  
53          EasyMock.replay(mockedResponse);
54  
55          // when
56          MessageStoreInterceptor msi = new MessageStoreInterceptor();
57          MessageStorePreResultListener listener = new MessageStorePreResultListener(msi);
58          listener.beforeResult(mockActionInvocation, Action.SUCCESS);
59  
60          // then
61          EasyMock.verify(mockActionInvocation);
62          EasyMock.verify(mockedRequest);
63          EasyMock.verify(mockedResponse);
64      }

97      public void testAutomatic() throws Exception {
98          MessageStoreInterceptor interceptor = new MessageStoreInterceptor();
99          interceptor.setOperationMode(MessageStoreInterceptor.AUTOMATIC_MODE);
100 
101         MessageStorePreResultListener listener = new MessageStorePreResultListener(interceptor);
102 
103         Map sessionMap = new LinkedHashMap();
104 
105         ActionSupport action = new ActionSupport();
106         action.addActionError("some action error 1");
107         action.addActionError("some action error 2");
108         action.addActionMessage("some action message 1");
109         action.addActionMessage("some action message 2");
110         action.addFieldError("field1", "some field error 1");
111         action.addFieldError("field2", "some field error 2");
112 
113         ActionContext actionContext = new ActionContext(new HashMap());
114         actionContext.setParameters(HttpParameters.create().build());
115         actionContext.put(ActionContext.SESSION, sessionMap);
116 
117         HttpSession mockedSession = EasyMock.createControl().createMock(HttpSession.class);
118         HttpServletRequest mockedRequest = EasyMock.createControl().createMock(HttpServletRequest.class);
119         mockedRequest.getSession(false);
120         EasyMock.expectLastCall().andReturn(mockedSession);
121         EasyMock.expectLastCall().once();
122         ServletActionContext.setRequest(mockedRequest);
123 
124         EasyMock.replay(mockedRequest);
125 
126         HttpServletResponse mockedResponse = EasyMock.createControl().createMock(HttpServletResponse.class);
127         mockedResponse.isCommitted();
128         EasyMock.expectLastCall().andReturn(false);
129         EasyMock.expectLastCall().once();
130         ServletActionContext.setResponse(mockedResponse);
131 
132         EasyMock.replay(mockedResponse);
133 
134         // Mock (ActionInvocation)
135         ActionInvocation mockActionInvocation = EasyMock.createControl().createMock(ActionInvocation.class);
136         mockActionInvocation.getInvocationContext();
137         EasyMock.expectLastCall().andReturn(actionContext);
138         EasyMock.expectLastCall().anyTimes();
139 
140         mockActionInvocation.getAction();
141         EasyMock.expectLastCall().andReturn(action);
142         EasyMock.expectLastCall().anyTimes();
143 
144         mockActionInvocation.getProxy();
145         MockActionProxy actionProxy = new MockActionProxy();
146         ResultConfig resultConfig = new ResultConfig.Builder(Action.SUCCESS, ServletRedirectResult.class.getName()).build();
147         ActionConfig actionConfig = new ActionConfig.Builder("", "test", action.getClass().getName()).addResultConfig(resultConfig).build();
148         actionProxy.setConfig(actionConfig);
149         EasyMock.expectLastCall().andReturn(actionProxy);
150         EasyMock.expectLastCall().anyTimes();
151 
152         EasyMock.replay(mockActionInvocation);
153

156 
157         List actionErrors = (List) sessionMap.get(MessageStoreInterceptor.actionErrorsSessionKey);
158         List actionMessages = (List) sessionMap.get(MessageStoreInterceptor.actionMessagesSessionKey);
159         Map fieldErrors = (Map) sessionMap.get(MessageStoreInterceptor.fieldErrorsSessionKey);
160 
161         assertEquals(actionErrors.size(), 2);
162         assertEquals(actionMessages.size(), 2);
163         assertEquals(fieldErrors.size(), 2);
164         assertTrue(actionErrors.contains("some action error 1"));

168         assertEquals(((List) fieldErrors.get("field1")).size(), 1);
169         assertEquals(((List) fieldErrors.get("field2")).size(), 1);
170         assertEquals(((List) fieldErrors.get("field1")).get(0), "some field error 1");
171         assertEquals(((List) fieldErrors.get("field2")).get(0), "some field error 2");
172 
173         EasyMock.verify(mockActionInvocation);
