54      // actionError
55      public void testActionErrorComponentDisposeItselfFromComponentStack() throws Exception {
56          ActionMessageTag t = new ActionMessageTag();
57          t.setPageContext(pageContext);
58  
59          try {
60              t.doStartTag();
61              ActionErrorTag tag = new ActionErrorTag();
62              tag.setPageContext(pageContext);
63              tag.doStartTag();
64              assertEquals(tag.getComponent().getComponentStack().peek(), tag.getComponent());
65              tag.doEndTag();
66              assertEquals(t.getComponent().getComponentStack().peek(), t.getComponent());
67  
68              t.doEndTag();
69          }
70          catch (Exception e) {
71              e.printStackTrace();
72              fail(e.toString());
73          }
74  
75      }
