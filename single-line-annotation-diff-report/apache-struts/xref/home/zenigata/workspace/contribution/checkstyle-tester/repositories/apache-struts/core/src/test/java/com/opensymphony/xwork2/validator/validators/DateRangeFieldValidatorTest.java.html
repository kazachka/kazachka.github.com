1   package com.opensymphony.xwork2.validator.validators;
2   
3   import com.opensymphony.xwork2.ActionContext;
4   import com.opensymphony.xwork2.TextProviderFactory;
5   import com.opensymphony.xwork2.XWorkTestCase;
6   import com.opensymphony.xwork2.util.ValueStack;
7   import com.opensymphony.xwork2.util.ValueStackFactory;
8   import com.opensymphony.xwork2.validator.DummyValidatorContext;
9   import com.opensymphony.xwork2.validator.ValidatorContext;
10  
11  import java.util.Calendar;
12  import java.util.Date;
13  import java.util.Locale;
14  
15  public class DateRangeFieldValidatorTest extends XWorkTestCase {
16  
17      private TextProviderFactory tpf;
18  
19      public void testPassValidation() throws Exception {
20          // given
21          ValidationAction action = prepareAction(createDate(2013, 6, 6));
22          ValidatorContext context = new DummyValidatorContext(action, tpf);
23          DateRangeFieldValidator validator = prepareValidator(action, context);
24  
25          // when
26          validator.validate(action);
27  
28          // then
29          assertTrue(context.getFieldErrors().size() == 0);
30      }
31  
32      public void testMinValidation() throws Exception {
33          // given
34          ValidationAction action = prepareAction(createDate(2012, Calendar.MARCH, 3));
35          ValidatorContext context = new DummyValidatorContext(action, tpf);
36          DateRangeFieldValidator validator = prepareValidator(action, context);
37  
38          // when
39          validator.validate(action);
40  
41          // then
42          assertTrue(context.getFieldErrors().size() == 1);
43          assertEquals("Max is 12.12.13, min is 01.01.13 but value is 03.03.12", context.getFieldErrors().get("dateRange").get(0));
44      }
45  
46      public void testMaxValidation() throws Exception {
47          // given
48          ValidationAction action = prepareAction(createDate(2014, Calendar.APRIL, 4));
49          ValidatorContext context = new DummyValidatorContext(action, tpf);
50          DateRangeFieldValidator validator = prepareValidator(action, context);
51  
52          // when
53          validator.validate(action);
54  
55          // then
56          assertTrue(context.getFieldErrors().size() == 1);
57          assertEquals("Max is 12.12.13, min is 01.01.13 but value is 04.04.14", context.getFieldErrors().get("dateRange").get(0));
58      }
59  
60      private ValidationAction prepareAction(Date range) {
61          ValidationAction action = new ValidationAction();
62          action.setDateMinValue(createDate(2013, Calendar.JANUARY, 1));
63          action.setDateMaxValue(createDate(2013, Calendar.DECEMBER, 12));
64          action.setDateRange(range);
65          return action;
66      }
67  
68      private Date createDate(int year, int month, int day) {
69          Calendar cal = Calendar.getInstance();
70          cal.set(year, month, day);
71          return cal.getTime();
72      }
73  
74      private DateRangeFieldValidator prepareValidator(ValidationAction action, ValidatorContext context) {
75          ValueStack valueStack = container.getInstance(ValueStackFactory.class).createValueStack();
76          valueStack.push(action);
77  
78          DateRangeFieldValidator validator = new DateRangeFieldValidator();
79          validator.setValueStack(valueStack);
80  
81          validator.setMaxExpression("${dateMaxValue}");
82          validator.setMinExpression("${dateMinValue}");
83          validator.setValidatorContext(context);
84          validator.setFieldName("dateRange");
85          validator.setDefaultMessage("Max is ${dateMaxValue}, min is ${dateMinValue} but value is ${dateRange}");
86  
87          return validator;
88      }
89  
90      @Override
91      public void setUp() throws Exception {
92          super.setUp();
93          ActionContext.getContext().setLocale(new Locale("DE"));
94          tpf = container.getInstance(TextProviderFactory.class);
95      }
96  
97  }
