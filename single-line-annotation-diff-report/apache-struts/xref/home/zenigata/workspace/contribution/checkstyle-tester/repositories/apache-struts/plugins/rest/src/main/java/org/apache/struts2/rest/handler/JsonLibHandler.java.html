22  package org.apache.struts2.rest.handler;
23  
24  import java.io.IOException;
25  import java.io.Reader;
26  import java.io.Writer;
27  import java.util.Collection;
28  
29  import net.sf.json.JSONArray;
30  import net.sf.json.JSONObject;
31  import net.sf.json.JsonConfig;
32  
33  import org.apache.struts2.StrutsConstants;
34  
35  import com.opensymphony.xwork2.inject.Inject;
36  
37  /**
38   * Handles JSON content using json-lib
39   */
40  public class JsonLibHandler implements ContentTypeHandler {
41  
42      private static final String DEFAULT_CONTENT_TYPE = "application/json";
43      private String defaultEncoding = "ISO-8859-1";
44  
45      public void toObject(Reader in, Object target) throws IOException {
46          StringBuilder sb = new StringBuilder();
47          char[] buffer = new char[1024];
48          int len = 0;
49          while ((len = in.read(buffer)) > 0) {
50              sb.append(buffer, 0, len);
51          }
52          if (target != null && sb.length() > 0 && sb.charAt(0) == '[') {
53              JSONArray jsonArray = JSONArray.fromObject(sb.toString());
54              if (target.getClass().isArray()) {
55                  JSONArray.toArray(jsonArray, target, new JsonConfig());
56              } else {
57                  JSONArray.toList(jsonArray, target, new JsonConfig());
58              }
59  
60          } else {
61              JSONObject jsonObject = JSONObject.fromObject(sb.toString());
62              JSONObject.toBean(jsonObject, target, new JsonConfig());
63          }
64      }
65  
66      public String fromObject(Object obj, String resultCode, Writer stream) throws IOException {
67          if (obj != null) {
68              if (isArray(obj)) {
69                  JSONArray jsonArray = JSONArray.fromObject(obj);
70                  stream.write(jsonArray.toString());
71              } else {
72                  JSONObject jsonObject = JSONObject.fromObject(obj);
73                  stream.write(jsonObject.toString());
74              }
75          }
76          return null;
77  
78  
79      }
80  
81      private boolean isArray(Object obj) {
82          return obj instanceof Collection || obj.getClass().isArray();
83      }
84  
85      public String getContentType() {
86          return DEFAULT_CONTENT_TYPE+";charset=" + this.defaultEncoding;
87      }
88      
89      public String getExtension() {
90          return "json";
91      }
92      
93      @Inject(StrutsConstants.STRUTS_I18N_ENCODING)
94      public void setDefaultEncoding(String val) {
95          this.defaultEncoding = val;
96      }
97  }
