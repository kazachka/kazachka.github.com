4   package com.opensymphony.xwork2.ognl.accessor;
5   
6   import ognl.ObjectPropertyAccessor;
7   import ognl.OgnlException;
8   import org.apache.struts2.dispatcher.Parameter;
9   
10  import java.util.Map;
11  
12  public class ParameterPropertyAccessor extends ObjectPropertyAccessor {
13  
14      @Override
15      public Object getProperty(Map context, Object target, Object oname) throws OgnlException {
16          if (target instanceof Parameter) {
17              if ("value".equalsIgnoreCase(String.valueOf(oname))) {
18                  throw new OgnlException("Access to " + oname + " is not allowed! Call parameter name directly!");
19              }
20              return ((Parameter) target).getObject();
21          }
22          return super.getProperty(context, target, oname);
23      }
24  
25      @Override
26      public void setProperty(Map context, Object target, Object oname, Object value) throws OgnlException {
27          if (target instanceof Parameter) {
28              throw new OgnlException("Access to " + target.getClass().getName() + " is read-only!");
29          } else {
30              super.setProperty(context, target, oname, value);
31          }
32      }
33  }
