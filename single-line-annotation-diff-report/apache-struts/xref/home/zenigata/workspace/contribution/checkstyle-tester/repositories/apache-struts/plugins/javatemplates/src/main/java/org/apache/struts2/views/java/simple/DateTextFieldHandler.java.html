1   package org.apache.struts2.views.java.simple;
2   
3   import java.io.IOException;
4   import java.text.SimpleDateFormat;
5   import java.util.Date;
6   import java.util.Map;
7   
8   import org.apache.struts2.interceptor.DateTextFieldInterceptor.DateWord;
9   import org.apache.struts2.views.java.Attributes;
10  import org.apache.struts2.views.java.TagGenerator;
11  
12  
13  public class DateTextFieldHandler extends AbstractTagHandler implements TagGenerator {
14  
15      @SuppressWarnings("unchecked")
16  	public void generate() throws IOException {
17          Map<String, Object> params = context.getParameters();
18          Attributes attr = null;
19  
20          // Get format
21          String format = (String)params.get("format");
22          String id = (String)params.get("id");
23          String name = (String)params.get("name");
24          if (id == null) {
25          	id = name;
26          }
27          Date date = (Date)params.get("nameValue");
28          
29          if (format != null) {
30          	// Verify if it's correct
31              new SimpleDateFormat(format);
32              
33              attr = new Attributes();
34              attr.addIfExists("id", id);
35              super.start("div", attr);
36              
37              Character antC = null;
38              for (Character c : format.toCharArray()) {
39              	
40              	try {
41              		DateWord dateWord = DateWord.get(c);
42              		if (!c.equals(antC)) {
43              			
44              			String cssClass = "date_" + dateWord.getDescription();
45              			if (params.get("cssClass") != null) {
46              				cssClass += " " + params.get("cssClass");
47              			}
48              			
49              			attr = new Attributes();
50              	        attr.add("type", "text")
51              	                .addIfExists("class", cssClass)
52              	                .addIfExists("size", dateWord.getLength())
53              	                .addIfExists("maxlength", dateWord.getLength())
54              	                .addIfTrue("disabled", params.get("disabled"))
55              	                .addIfTrue("readonly", params.get("readonly"))
56              	                .addIfExists("tabindex", params.get("tabindex"))
57              	                .addIfExists("style", params.get("cssStyle"))
58              	                .addIfExists("title", params.get("title"));
59              	        
60              	        if (id != null && !"".equals(id)) {
61                  	        attr.addDefaultToEmpty("id", "__" + dateWord.getDescription() + "_" + id);
62              	        }
63              	        if (name != null && !"".equals(id)) {
64                  	        attr.addDefaultToEmpty("name", "__" + dateWord.getDescription() + "_" + name);
65              	        } else {
66                  	        attr.addDefaultToEmpty("name", dateWord.getDescription());
67              	        }
68              	        if (date != null) {
69              	        	SimpleDateFormat formatter = new SimpleDateFormat(dateWord.getDateType());
70              	        	attr.addIfExists("value", formatter.format(date), false);
71              	        }
72              	        
73              	        super.start("input", attr);
74              	        super.end("input");
75              	        
76              		}
77              	} catch (IllegalArgumentException e) {
78              		super.characters(c.toString());	
79              	}
80              	antC = c;
81              
82              }
83              super.end("div");
84          }
85          
86      }
87      
88      
89      
90  }
