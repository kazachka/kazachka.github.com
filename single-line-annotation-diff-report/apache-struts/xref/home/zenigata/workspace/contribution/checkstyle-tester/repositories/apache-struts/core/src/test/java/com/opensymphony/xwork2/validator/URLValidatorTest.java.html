114 		URLValidator validator = new URLValidator();
115 
116         validator.setUrlRegex("^myapp:\\/\\/[a-z]*\\.com$");
117 
118         Pattern pattern = Pattern.compile(validator.getUrlRegex());
119 
120         assertTrue(pattern.matcher("myapp://test.com").matches());
121         assertFalse(pattern.matcher("myap://test.com").matches());

136 	public void testValidUrlWithDefaultRegex() throws Exception {
137 		URLValidator validator = new URLValidator();
138 
139         Pattern pattern = Pattern.compile(validator.getUrlRegex(), Pattern.CASE_INSENSITIVE);
140 
141         assertFalse(pattern.matcher("myapp://test.com").matches());
142         assertFalse(pattern.matcher("myap://test.com").matches());
143         assertFalse(pattern.matcher("").matches());
144         assertFalse(pattern.matcher("   ").matches());
145         assertFalse(pattern.matcher("no url").matches());
146 		assertFalse(pattern.matcher("http://example.com////////////////////////////////////////////////////////////////////////////////////??").matches());
147 
148         assertTrue(pattern.matcher("http://www.opensymphony.com").matches());
149         assertTrue(pattern.matcher("https://www.opensymphony.com").matches());
150         assertTrue(pattern.matcher("https://www.opensymphony.com:443/login").matches());
151         assertTrue(pattern.matcher("http://localhost:8080/myapp").matches());
152 
153         assertTrue(pattern.matcher("http://www.legalspace.com/__media__/js/netsoltrademark.php?d=www.a-vos-travaux.fr%2Facheter-un-aspirateur-sans-sac-pas-cher%2F").matches());
154         assertTrue(UrlValidator.getInstance().isValid("http://www.legalspace.com/__media__/js/netsoltrademark.php?d=www.a-vos-travaux.fr%2Facheter-un-aspirateur-sans-sac-pas-cher%2F"));
155 
156 		assertTrue(pattern.matcher("http://www.duadmin.isaev.Infoduadmin.Isaev.info/?a%5B%5D=%3Ca%20href%3Dhttp%3A%2F%2Fwww.aspert.fr%2Fun-seche-cheveux-lisseur-est-il-vraiment-utile%2F%3Eseche%20cheveux%20dyson%20test%3C%2Fa").matches());
157 		assertTrue(UrlValidator.getInstance().isValid("http://www.duadmin.isaev.Infoduadmin.Isaev.info/?a%5B%5D=%3Ca%20href%3Dhttp%3A%2F%2Fwww.aspert.fr%2Fun-seche-cheveux-lisseur-est-il-vraiment-utile%2F%3Eseche%20cheveux%20dyson%20test%3C%2Fa"));
158 
159 		assertTrue(pattern.matcher("http://netsol-underconstruction-page-monitor-1.com/__media__/js/netsoltrademark.php?d=www.le-soutien-scolaire.fr%2Favis-et-test-comparatifs-des-robots-multifonctions%2F").matches());
160 		assertTrue(UrlValidator.getInstance().isValid("http://netsol-underconstruction-page-monitor-1.com/__media__/js/netsoltrademark.php?d=www.le-soutien-scolaire.fr%2Favis-et-test-comparatifs-des-robots-multifonctions%2F"));

163 	public void testValidUrlCaseInsensitive() throws Exception {
164 		// given
165 		final Map<String, Object> fieldErrors = new HashMap<>();
166 
167 		URLValidator validator = new URLValidator() {
168 			@Override
169 			public String getFieldName() {
170 				return "url";
171 			}
172 
173 			@Override
174 			protected Object getFieldValue(String name, Object object) throws ValidationException {
175 				return object;
176 			}
177 
178 			@Override
179 			protected void addFieldError(String propertyName, Object object) {
180 				fieldErrors.put(propertyName, object);
181 			}
182 		};
183 
184 		// when
185         validator.validate("http://localhost:8080/myapp");
186 
187 		// then
188 		assertTrue(fieldErrors.isEmpty());
189 
190 		// when
191         validator.validate("http://LOCALHOST:8080/MYAPP");
192 
193 		// then
194 		assertTrue(fieldErrors.isEmpty());
195 
196 		// when
197         validator.validate("http://www.appache.org/TEST");
198 
199 		// then
200 		assertTrue(fieldErrors.isEmpty());
201     }
