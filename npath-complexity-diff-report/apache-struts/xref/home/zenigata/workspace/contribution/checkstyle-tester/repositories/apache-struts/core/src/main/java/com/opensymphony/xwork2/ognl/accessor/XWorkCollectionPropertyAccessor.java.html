76      /**
77       * Gets the property of a Collection by indexing the collection
78       * based on a key property. For example, if the key property were
79       * 'id', this method would convert the key Object to whatever
80       * type the id property was, and then access the Set like it was
81       * a Map returning a JavaBean with the value of id property matching
82       * the input.
83       *
84       * @see ognl.PropertyAccessor#getProperty(java.util.Map, Object, Object)
85       */
86      @Override
87      public Object getProperty(Map context, Object target, Object key) throws OgnlException {
88          LOG.trace("Entering getProperty()");
89  
90          //check if it is a generic type property.
91          //if so, return the value from the
92          //superclass which will determine this.
93          if (!ReflectionContextState.isGettingByKeyProperty(context) && !key.equals(KEY_PROPERTY_FOR_CREATION)) {
94              return super.getProperty(context, target, key);
95          }	else {
96              //reset context property
97              ReflectionContextState.setGettingByKeyProperty(context,false);
98          }
99          Collection c = (Collection) target;
100 
101         //get the bean that this collection is a property of
102         Class lastBeanClass = ReflectionContextState.getLastBeanClassAccessed(context);
103 
104         //get the property name that this collection uses
105         String lastPropertyClass = ReflectionContextState.getLastBeanPropertyAccessed(context);
106 
107         //if one or the other is null, assume that it isn't
108         //set up correctly so just return whatever the
109         //superclass would
110         if (lastBeanClass == null || lastPropertyClass == null) {
111             ReflectionContextState.updateCurrentPropertyPath(context, key);
112             return super.getProperty(context, target, key);
113         }
114 
115         //get the key property to index the
116         //collection with from the ObjectTypeDeterminer
117         String keyProperty = objectTypeDeterminer.getKeyProperty(lastBeanClass, lastPropertyClass);
118 
119         //get the collection class of the
120         Class collClass = objectTypeDeterminer.getElementClass(lastBeanClass, lastPropertyClass, key);
121 
122         Class keyType;
123         Class toGetTypeFrom = (collClass != null) ? collClass : c.iterator().next().getClass();
124         try {
125             keyType = OgnlRuntime.getPropertyDescriptor(toGetTypeFrom, keyProperty).getPropertyType();
126         } catch (Exception exc) {
127             throw new OgnlException("Error getting property descriptor: " + exc.getMessage());
128         }
129 
130 
131         if (ReflectionContextState.isCreatingNullObjects(context)) {
132             Map collMap = getSetMap(context, c, keyProperty);
133             if (key.toString().equals(KEY_PROPERTY_FOR_CREATION)) {
134                 //this should return the XWorkList
135                 //for this set that contains new entries
136                 //then the ListPropertyAccessor will be called
137                 //to access it in the next sequence
138                 return collMap.get(null);
139             }
140             Object realKey = xworkConverter.convertValue(context, key, keyType);
141             Object value = collMap.get(realKey);
142             if (value == null
143                     && ReflectionContextState.isCreatingNullObjects(context)
144                     && objectTypeDeterminer
145                     .shouldCreateIfNew(lastBeanClass,lastPropertyClass,c,keyProperty,false)) {
146                 	//create a new element and 
147                     //set the value of keyProperty
148                     //to be the given value
149                 	try {
150                 	    value=objectFactory.buildBean(collClass, context);
151                 	    
152                 	    //set the value of the keyProperty
153                 	    _accessor.setProperty(context,value,keyProperty,realKey);
154                 	    
155                 	    //add the new object to the collection 
156                 	    c.add(value);
157                 	    
158                 	    //add to the Map if accessed later
159                 	    collMap.put(realKey, value);
160                 	    
161                 	    
162                 	}	catch (Exception exc) {
163                 	    throw new OgnlException("Error adding new element to collection", exc);
164                 	}
165                 
166             }
167             return value;
168         } else {
169             if (key.toString().equals(KEY_PROPERTY_FOR_CREATION)) {
170                 return null;
171             }
172             //with getting do iteration
173             //don't assume for now it is
174             //optimized to create the Map
175             //and unlike setting, there is
176             //no easy key for the Set
177             Object realKey = xworkConverter.convertValue(context, key, keyType);
178             return getPropertyThroughIteration(context, c, keyProperty, realKey);
179         }
180     }

182     /*
183       * Gets an indexed Map by a given key property with the key being
184       * the value of the property and the value being the
185       */
186     private Map getSetMap(Map context, Collection collection, String property) throws OgnlException {
187         LOG.trace("getting set Map");
188 
189         String path = ReflectionContextState.getCurrentPropertyPath(context);
190         Map map = ReflectionContextState.getSetMap(context, path);
191 
192         if (map == null) {
193             LOG.trace("creating set Map");
194 
195             map = new HashMap();
196             map.put(null, new SurrugateList(collection));
197             for (Object currTest : collection) {
198                 Object currKey = _accessor.getProperty(context, currTest, property);
199                 if (currKey != null) {
200                     map.put(currKey, currTest);
201                 }
202             }
203             ReflectionContextState.setSetMap(context, map, path);
204         }
205         return map;
206     }

223     @Override
224     public void setProperty(Map context, Object target, Object name, Object value) throws OgnlException {
225         Class lastClass = (Class) context.get(XWorkConverter.LAST_BEAN_CLASS_ACCESSED);
226         String lastProperty = (String) context.get(XWorkConverter.LAST_BEAN_PROPERTY_ACCESSED);
227         Class convertToClass = objectTypeDeterminer.getElementClass(lastClass, lastProperty, name);
228 
229         if (name instanceof String && value.getClass().isArray()) {
230             // looks like the input game in the form of "someCollection.foo" and
231             // we are expected to define the index values ourselves.
232             // So let's do it:
233 
234             Collection c = (Collection) target;
235             Object[] values = (Object[]) value;
236             for (Object v : values) {
237                 try {
238                     Object o = objectFactory.buildBean(convertToClass, context);
239                     ognlUtil.setValue((String) name, context, o, v);
240                     c.add(o);
241                 } catch (Exception e) {
242                     throw new OgnlException("Error converting given String values for Collection.", e);
243                 }
244             }
245 
246             // we don't want to do the normal collection property setting now, since we've already done the work
247             // just return instead
248             return;
249         }
250 
251         Object realValue = getRealValue(context, value, convertToClass);
252 
253         super.setProperty(context, target, name, realValue);
254     }
255 
256     private Object getRealValue(Map context, Object value, Class convertToClass) {
257         if (value == null || convertToClass == null) {
258             return value;
259         }
260         return xworkConverter.convertValue(context, value, convertToClass);
261     }
