3   import com.opensymphony.xwork2.conversion.impl.DefaultTypeConverter;
4   
5   import java.util.Map;
6   
7   public class AddressTypeConverter extends DefaultTypeConverter {
8   	@Override public Object convertValue(Map<String, Object> context, Object value, Class toType) {
9   		if(value instanceof String) {
10  			return decodeAddress((String)value);
11  		} else if(value instanceof String && value.getClass().isArray()) {
12  			return decodeAddress(((String[])value)[0]);
13  		} else {
14  			Address address = (Address)value;
15  			return address.getLine1() + ":" + address.getLine2() + ":" +
16  			       address.getCity() + ":" + address.getCountry();
17  		}
18  	}
19  
20  	private Address decodeAddress(String encodedAddress) {
21  		String[] parts = ((String)encodedAddress).split(":");
22  		Address address = new Address();
23  		address.setLine1(parts[0]);
24  		address.setLine2(parts[1]);
25  		address.setCity(parts[2]);
26  		address.setCountry(parts[3]);
27  		return address;
28  	}
29  }
