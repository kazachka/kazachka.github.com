3   import com.opensymphony.xwork2.ObjectFactory;
4   import com.opensymphony.xwork2.config.ConfigurationException;
5   import com.opensymphony.xwork2.config.entities.InterceptorConfig;
6   import com.opensymphony.xwork2.inject.Inject;
7   import com.opensymphony.xwork2.interceptor.Interceptor;
8   import com.opensymphony.xwork2.interceptor.WithLazyParams;
9   import com.opensymphony.xwork2.util.reflection.ReflectionProvider;
10  import org.apache.logging.log4j.LogManager;
11  import org.apache.logging.log4j.Logger;
12  
13  import java.util.HashMap;
14  import java.util.Map;
15  
16  /**
17   * Default implementation
18   */
19  public class DefaultInterceptorFactory implements InterceptorFactory {
20  
21      private static final Logger LOG = LogManager.getLogger(DefaultInterceptorFactory.class);
22  
23      private ObjectFactory objectFactory;
24      private ReflectionProvider reflectionProvider;
25  
26      @Inject
27      public void setObjectFactory(ObjectFactory objectFactory) {
28          this.objectFactory = objectFactory;
29      }
30  
31      @Inject
32      public void setReflectionProvider(ReflectionProvider reflectionProvider) {
33          this.reflectionProvider = reflectionProvider;
34      }
35  
36      public Interceptor buildInterceptor(InterceptorConfig interceptorConfig, Map<String, String> interceptorRefParams) throws ConfigurationException {
37          String interceptorClassName = interceptorConfig.getClassName();
38          Map<String, String> thisInterceptorClassParams = interceptorConfig.getParams();
39          Map<String, String> params = (thisInterceptorClassParams == null) ? new HashMap<String, String>() : new HashMap<>(thisInterceptorClassParams);
40          params.putAll(interceptorRefParams);
41  
42          String message;
43          Throwable cause;
44  
45          try {
46              // interceptor instances are long-lived and used across user sessions, so don't try to pass in any extra context
47              Object o = objectFactory.buildBean(interceptorClassName, null);
48              if (o instanceof WithLazyParams) {
49                  LOG.debug("Interceptor {} is marked with interface {} and params will be set during action invocation",
50                          interceptorClassName, WithLazyParams.class.getName());
51              } else {
52                  reflectionProvider.setProperties(params, o);
53              }
54  
55              if (o instanceof Interceptor) {
56                  Interceptor interceptor = (Interceptor) o;
57                  interceptor.init();
58                  return interceptor;
59              }
60  
61              throw new ConfigurationException("Class [" + interceptorClassName + "] does not implement Interceptor", interceptorConfig);
62          } catch (InstantiationException e) {
63              cause = e;
64              message = "Unable to instantiate an instance of Interceptor class [" + interceptorClassName + "].";
65          } catch (IllegalAccessException e) {
66              cause = e;
67              message = "IllegalAccessException while attempting to instantiate an instance of Interceptor class [" + interceptorClassName + "].";
68          } catch (ClassCastException e) {
69              cause = e;
70              message = "Class [" + interceptorClassName + "] does not implement com.opensymphony.xwork2.interceptor.Interceptor";
71          } catch (Exception e) {
72              cause = e;
73              message = "Caught Exception while registering Interceptor class " + interceptorClassName;
74          } catch (NoClassDefFoundError e) {
75              cause = e;
76              message = "Could not load class " + interceptorClassName + ". Perhaps it exists but certain dependencies are not available?";
77          }
78  
79          throw new ConfigurationException(message, cause, interceptorConfig);
80      }
81  
82  }
